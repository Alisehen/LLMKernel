import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def maxpool_hardtanh_kernel(
    x_ptr, y_ptr,
    N, C, H, W,
    H_out, W_out,
    stride_xn, stride_xc, stride_xh, stride_xw,
    stride_yn, stride_yc, stride_yh, stride_yw,
    kH, kW,
    sH, sW,
    hardtanh_min, hardtanh_max,
    BLOCK_P: tl.constexpr,
):
    """
    Fused 2D MaxPool + Hardtanh.

    Input:  x[N, C, H, W]
    Output: y[N, C, H_out, W_out] where
        H_out = floor((H - kH) / sH) + 1
        W_out = floor((W - kW) / sW) + 1
    """
    pid_nc = tl.program_id(0)
    pid_p = tl.program_id(1)

    total_nc = N * C
    if pid_nc >= total_nc:
        return

    n = pid_nc // C
    c = pid_nc % C

    total_P = H_out * W_out
    offs_p = pid_p * BLOCK_P + tl.arange(0, BLOCK_P)
    mask_p = offs_p < total_P

    # Spatial indices of pooled output
    oh = offs_p // W_out
    ow = offs_p % W_out

    base_y = oh * sH
    base_x = ow * sW

    # Initialize max values with -inf
    max_vals = tl.full((BLOCK_P,), -float("inf"), dtype=tl.float32)

    # Max pooling window
    for kh in range(0, kH):
        for kw in range(0, kW):
            iy = base_y + kh
            ix = base_x + kw
            x_ptrs = (
                x_ptr
                + n * stride_xn
                + c * stride_xc
                + iy * stride_xh
                + ix * stride_xw
            )
            vals = tl.load(x_ptrs, mask=mask_p, other=-float("inf"))
            max_vals = tl.maximum(max_vals, vals)

    # Hardtanh clamp
    clamped = tl.minimum(tl.maximum(max_vals, hardtanh_min), hardtanh_max)

    # Store pooled + hardtanh result
    y_ptrs = (
        y_ptr
        + n * stride_yn
        + c * stride_yc
        + oh * stride_yh
        + ow * stride_yw
    )
    tl.store(y_ptrs, clamped, mask=mask_p)


@triton.jit
def mean_tanh_kernel(
    x_ptr, out_ptr,
    N, C, H, W,
    stride_xn, stride_xc, stride_xh, stride_xw,
    stride_on, stride_oc,
    BLOCK_P: tl.constexpr,
):
    """
    Global mean over H, W followed by tanh.

    Input:  x[N, C, H, W]
    Output: out[N, C, 1, 1]
    """
    pid_nc = tl.program_id(0)
    total_nc = N * C
    if pid_nc >= total_nc:
        return

    n = pid_nc // C
    c = pid_nc % C

    total_P = H * W
    acc = 0.0

    # Reduce over all spatial positions in blocks of BLOCK_P
    for p_start in range(0, total_P, BLOCK_P):
        offs = p_start + tl.arange(0, BLOCK_P)
        mask = offs < total_P

        oh = offs // W
        ow = offs % W

        x_ptrs = (
            x_ptr
            + n * stride_xn
            + c * stride_xc
            + oh * stride_xh
            + ow * stride_xw
        )
        vals = tl.load(x_ptrs, mask=mask, other=0.0)
        acc += tl.sum(vals, axis=0)

    mean_val = acc / total_P

    # tanh(mean) using exp: tanh(x) = (e^{2x} - 1) / (e^{2x} + 1)
    two_mean = 2.0 * mean_val
    e2x = tl.exp(two_mean)
    tanh_val = (e2x - 1.0) / (e2x + 1.0)

    out_ptrs = out_ptr + n * stride_on + c * stride_oc
    tl.store(out_ptrs, tanh_val)


def fused_post_convtranspose(
    x,
    maxpool_kernel_size,
    maxpool_stride,
    hardtanh_min,
    hardtanh_max,
):
    """
    Fused:
        MaxPool2d + Hardtanh + GlobalMean (H,W, keepdim=True) + tanh

    x: [N, C, H, W] (output of ConvTranspose2d)
    returns: [N, C, 1, 1]
    """
    assert x.is_cuda, "Input must be on CUDA for Triton kernels."

    N, C, H, W = x.shape

    # Normalize kernel_size and stride to (kH, kW), (sH, sW)
    if isinstance(maxpool_kernel_size, int):
        kH = kW = maxpool_kernel_size
    else:
        kH, kW = maxpool_kernel_size

    if isinstance(maxpool_stride, int):
        sH = sW = maxpool_stride
    else:
        sH, sW = maxpool_stride

    # Compute pooled output spatial size (no padding)
    H_out = (H - kH) // sH + 1
    W_out = (W - kW) // sW + 1

    # Intermediate buffer after maxpool + hardtanh
    y = torch.empty((N, C, H_out, W_out), device=x.device, dtype=x.dtype)
    # Final output [N, C, 1, 1]
    out = torch.empty((N, C, 1, 1), device=x.device, dtype=x.dtype)

    BLOCK_P1 = 64   # power-of-two
    BLOCK_P2 = 128  # power-of-two

    total_P = H_out * W_out

    # Kernel 1: MaxPool + Hardtanh
    grid1 = (N * C, triton.cdiv(total_P, BLOCK_P1))
    maxpool_hardtanh_kernel[grid1](
        x, y,
        N, C, H, W,
        H_out, W_out,
        x.stride(0), x.stride(1), x.stride(2), x.stride(3),
        y.stride(0), y.stride(1), y.stride(2), y.stride(3),
        kH, kW,
        sH, sW,
        hardtanh_min, hardtanh_max,
        BLOCK_P=BLOCK_P1,
    )

    # Kernel 2: Global mean over H,W + tanh
    grid2 = (N * C,)
    mean_tanh_kernel[grid2](
        y, out,
        N, C, H_out, W_out,
        y.stride(0), y.stride(1), y.stride(2), y.stride(3),
        out.stride(0), out.stride(1),
        BLOCK_P=BLOCK_P2,
    )

    return out


class ModelNew(nn.Module):
    """
    ConvTranspose2d (PyTorch native) +
    fused Triton kernels implementing:
        MaxPool2d + Hardtanh + GlobalMean (H,W, keepdim=True) + tanh
    """
    def __init__(
        self,
        in_channels,
        out_channels,
        kernel_size,
        stride,
        padding,
        maxpool_kernel_size,
        maxpool_stride,
        hardtanh_min,
        hardtanh_max,
    ):
        super(ModelNew, self).__init__()
        # ConvTranspose2d stays as PyTorch native (indexing is complex)
        self.conv_transpose = nn.ConvTranspose2d(
            in_channels,
            out_channels,
            kernel_size,
            stride=stride,
            padding=padding,
        )

        # Store pooling / activation parameters for Triton part
        if isinstance(maxpool_kernel_size, int):
            self.maxpool_kernel_size = (maxpool_kernel_size, maxpool_kernel_size)
        else:
            self.maxpool_kernel_size = tuple(maxpool_kernel_size)

        if isinstance(maxpool_stride, int):
            self.maxpool_stride = (maxpool_stride, maxpool_stride)
        else:
            self.maxpool_stride = tuple(maxpool_stride)

        self.hardtanh_min = float(hardtanh_min)
        self.hardtanh_max = float(hardtanh_max)

    def forward(self, x):
        # ConvTranspose2d in PyTorch
        x = self.conv_transpose(x)
        # Fused MaxPool2d + Hardtanh + Mean + tanh in Triton
        x = fused_post_convtranspose(
            x,
            self.maxpool_kernel_size,
            self.maxpool_stride,
            self.hardtanh_min,
            self.hardtanh_max,
        )
        return x
