import torch, torch.nn as nn, triton, triton.language as tl


@triton.jit
def sum_hw_kernel(
    x_ptr, out_ptr,
    B, C, HW,
    stride_b, stride_c,
    BLOCK_HW: tl.constexpr,
):
    pid = tl.program_id(0)
    b = pid // C
    c = pid % C

    # pointer to (b, c, 0, 0)
    base_offset = b * stride_b + c * stride_c
    base_ptr = x_ptr + base_offset

    acc = 0.0
    offs = tl.arange(0, BLOCK_HW)
    for start in range(0, HW, BLOCK_HW):
        idx = start + offs
        mask = idx < HW
        vals = tl.load(base_ptr + idx, mask=mask, other=0.0)
        acc += tl.sum(vals, axis=0)

    tl.store(out_ptr + pid, acc)


@triton.jit
def fused_gemm_mean_bias_logsumexp_10_kernel(
    a_ptr,        # [M, K]
    w_ptr,        # [K, N]
    bias_ptr,     # [N]
    out_ptr,      # [M]
    M, N, K,
    stride_am, stride_ak,
    stride_bk, stride_bn,
    stride_out,
    scale,
    BLOCK_M: tl.constexpr, BLOCK_N: tl.constexpr, BLOCK_K: tl.constexpr,
):
    """
    For each row m in [0, M):
        g[m, :] = (a[m, :] @ w[:, :]) * scale + bias[:]
        out[m]  = 10 * logsumexp(g[m, :], dim=0)

    Computes GEMM in tiles over N and K, and performs an online, numerically
    stable log-sum-exp over N while accumulating, so g is never written to DRAM.
    """
    pid_m = tl.program_id(0)

    offs_m = pid_m * BLOCK_M + tl.arange(0, BLOCK_M)
    mask_m = offs_m < M

    # Running log-sum-exp state per row: max_row, sum_row
    neg_inf = -float("inf")
    max_row = tl.full((BLOCK_M,), neg_inf, dtype=tl.float32)
    sum_row = tl.zeros((BLOCK_M,), dtype=tl.float32)

    offs_k = tl.arange(0, BLOCK_K)

    # Loop over N in tiles
    for n_start in range(0, N, BLOCK_N):
        offs_n = n_start + tl.arange(0, BLOCK_N)
        mask_n = offs_n < N

        # Accumulator for this (M, N_tile) block
        acc = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)

        # Pointers for K-loop for this N-tile
        a_ptrs = a_ptr + offs_m[:, None] * stride_am + offs_k[None, :] * stride_ak
        b_ptrs = w_ptr + offs_k[:, None] * stride_bk + offs_n[None, :] * stride_bn

        # Loop over K
        for k_start in range(0, K, BLOCK_K):
            k_remaining = K - k_start
            k_mask = offs_k < k_remaining

            a = tl.load(
                a_ptrs,
                mask=mask_m[:, None] & k_mask[None, :],
                other=0.0,
            )
            b = tl.load(
                b_ptrs,
                mask=k_mask[:, None] & mask_n[None, :],
                other=0.0,
            )
            acc += tl.dot(a, b, allow_tf32=True)

            a_ptrs += BLOCK_K * stride_ak
            b_ptrs += BLOCK_K * stride_bk

        # Scale and add bias
        acc = acc * scale
        bias = tl.load(bias_ptr + offs_n, mask=mask_n, other=0.0)
        acc += bias[None, :]

        # Mask invalid rows/cols for max and sum
        acc_for_max = tl.where(
            mask_n[None, :] & mask_m[:, None],
            acc,
            neg_inf,
        )

        # Tile-wise max per row
        tile_max = tl.max(acc_for_max, axis=1)

        # Update running max
        new_max = tl.maximum(max_row, tile_max)

        # Compute tile contribution to sum_exp with respect to new_max
        acc_shifted = acc_for_max - new_max[:, None]
        exp_shifted = tl.exp(acc_shifted)
        tile_sum = tl.sum(exp_shifted, axis=1)

        # Scale previous sum_row to new_max, avoiding NaNs when max_row=-inf
        has_prev = max_row > neg_inf
        scale_prev = tl.where(
            has_prev,
            tl.exp(max_row - new_max),
            0.0,
        )
        sum_row = sum_row * scale_prev + tile_sum
        max_row = new_max

    # Final log-sum-exp per row and multiply by 10
    log_sum = tl.log(sum_row)
    result = (log_sum + max_row) * 10.0

    tl.store(out_ptr + offs_m * stride_out, result, mask=mask_m)


def triton_sum_hw(x: torch.Tensor) -> torch.Tensor:
    """
    x: [B, C, H, W] (NCHW, contiguous)
    returns: [B, C] with sum over H,W
    """
    x = x.contiguous()
    B, C, H, W = x.shape
    HW = H * W
    out = torch.empty(B * C, device=x.device, dtype=x.dtype)

    stride_b = x.stride(0)
    stride_c = x.stride(1)

    grid = (B * C,)
    sum_hw_kernel[grid](
        x, out,
        B, C, HW,
        stride_b, stride_c,
        BLOCK_HW=256,
    )
    return out.view(B, C)


def triton_fused_gemm_mean_bias_logsumexp_10(
    a: torch.Tensor,          # [B, C_in] = sum over H,W of input
    weight_sum: torch.Tensor, # [C_in, C_out] = sum over kh,kw of weight
    conv_bias: torch.Tensor,  # [C_out]
    bias2: torch.Tensor,      # [C_out, 1, 1]
    H: int, W: int,
    kH: int, kW: int,
) -> torch.Tensor:
    """
    Computes:
        g[b, c_out] = mean_HW(convT(x))[b, c_out] + bias2[c_out]

    but instead of materializing g[b, c_out], it directly returns:
        out[b, 0] = 10 * logsumexp_c_out(g[b, c_out])

    using the algebraic reduction:
        mean_HW(convT) = (sum_HW(x) @ weight_sum) / (H_out*W_out) + conv_bias
    """
    B, C_in = a.shape
    C_in_w, C_out = weight_sum.shape
    assert C_in_w == C_in

    H_out = H + kH - 1
    W_out = W + kW - 1
    scale = 1.0 / float(H_out * W_out)

    # Bias after pooling (conv bias) + separate bias parameter
    bias2_flat = bias2.view(-1)
    bias_total = conv_bias + bias2_flat  # [C_out]

    a = a.contiguous()
    w = weight_sum.contiguous()
    bias_total = bias_total.contiguous()

    out = torch.empty(B, device=a.device, dtype=a.dtype)

    grid = lambda META: (
        triton.cdiv(B, META["BLOCK_M"]),
    )

    fused_gemm_mean_bias_logsumexp_10_kernel[grid](
        a, w, bias_total, out,
        B, C_out, C_in,
        a.stride(0), a.stride(1),
        w.stride(0), w.stride(1),
        out.stride(0),
        scale,
        BLOCK_M=16, BLOCK_N=64, BLOCK_K=32,
    )
    return out.view(B, 1)


class ModelNew(nn.Module):
    """
    Triton-optimized version of the original model.

    Uses algebraic simplification of ConvTranspose2d + global average pooling:
    Instead of forming the full [B, C_out, H_out, W_out] tensor, it computes
    the pooled values directly from sums of inputs and kernel weights, and
    fuses the GEMM, bias addition, log-sum-exp, and final scaling into a
    single Triton kernel.
    """

    def __init__(self, in_channels, out_channels, kernel_size, bias_shape):
        super(ModelNew, self).__init__()
        if isinstance(kernel_size, int):
            kH = kW = kernel_size
        else:
            kH, kW = kernel_size
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kH = kH
        self.kW = kW

        # ConvTranspose2d-like parameters: [in_channels, out_channels, kH, kW]
        self.weight = nn.Parameter(
            torch.randn(in_channels, out_channels, kH, kW)
        )
        self.conv_bias = nn.Parameter(torch.randn(out_channels))

        # Additional bias added after global average pooling
        self.bias = nn.Parameter(torch.randn(bias_shape))

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # x: [B, C_in, H, W]
        B, C_in, H, W = x.shape
        assert C_in == self.in_channels

        # 1) Sum over spatial dims (H, W)
        sum_x = triton_sum_hw(x)  # [B, C_in]

        # 2) Sum weights over kernel dims (kh, kw): [C_in, C_out]
        weight_sum = self.weight.sum(dim=(2, 3))

        # 3) Fused GEMM + scaling + bias + logsumexp * 10
        out = triton_fused_gemm_mean_bias_logsumexp_10(
            sum_x,
            weight_sum,
            self.conv_bias,
            self.bias,
            H, W,
            self.kH, self.kW,
        )  # [B, 1]

        return out
