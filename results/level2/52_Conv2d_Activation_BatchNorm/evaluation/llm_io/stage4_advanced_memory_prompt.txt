def fused_activation_kernel(
    x_ptr,
    output_ptr,
    n_elements,
    BLOCK_SIZE: tl.constexpr,
    VECTORIZE: tl.constexpr,
):
    pid = tl.program_id(axis=0)
    block_start = pid * BLOCK_SIZE * VECTORIZE
    
    # Vectorized offsets for better memory coalescing
    offsets = block_start + tl.arange(0, BLOCK_SIZE)[:, None] * VECTORIZE + tl.arange(0, VECTORIZE)[None, :]
    mask = offsets < n_elements
    
    # Vectorized load
    x = tl.load(x_ptr + offsets, mask=mask, other=0.0)
    
    # Optimized softplus computation
    # For x > 0: x + log(1 + exp(-x))
    # For x <= 0: log(1 + exp(x))
    is_positive = x > 0
    neg_x = -x
    exp_neg_x = tl.where(x < 10.0, tl.exp(neg_x), 0.0)
    exp_x = tl.where(x > -10.0, tl.exp(x), 0.0)
    
    # Compute log(1 + exp_neg_x) and log(1 + exp_x) without log1p
    log1p_exp_neg_x = tl.where(exp_neg_x < 1e-4, 
                               -x - 0.5 * exp_neg_x,
                               tl.log(1.0 + exp_neg_x))
    
    log1p_exp_x = tl.where(exp_x < 1e-4,
                           x - 0.5 * exp_x,
                           tl.log(1.0 + exp_x))
    
    y = tl.where(is_positive, x + log1p_exp_neg_x, log1p_exp_x)

    y2 = y * y

                     tl.where(y > 0, 1.0, -1.0))
    
    # Final output
    output = x * tanh_y
    
    # Vectorized store
    tl.store(output_ptr + offsets, output, mask=mask)

    offsets = offsets_base + range_vec[None, :]

    large_negative = x < -threshold
    
    # For moderate values, compute tanh(softplus(x))
    # Use the identity: tanh(y) = (exp(2y) - 1) / (exp(2y) + 1)
    # where y = softplus(x)
    
    # Compute softplus(x) = log(1 + exp(x))
    # Use log(1 + exp(x)) instead of log1p(exp(x))
    exp_x = tl.exp(x)
    softplus_x = tl.where(x > 0, x + tl.log(1.0 + tl.exp(-x)), tl.log(1.0 + exp_x))
    
    # Compute tanh(softplus_x)
    exp_2y = tl.exp(2.0 * softplus_x)

    # For large negative x: softplus(x) ≈ 0, tanh(0) = 0, so output ≈ 0

                     tl.where(large_negative, 0.0,
                             x * tanh_val))
    
    # Store result
    tl.store(output_ptr + offsets, result, mask=mask)
