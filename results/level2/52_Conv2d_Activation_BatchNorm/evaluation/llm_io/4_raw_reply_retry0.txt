@triton.jit
def fused_activation_kernel(
    x_ptr,
    output_ptr,
    n_elements,
    BLOCK_SIZE: tl.constexpr,
    VECTORIZE: tl.constexpr,
    USE_TF32: tl.constexpr,
):

    block_start = pid * BLOCK_SIZE * VECTORIZE

        x = tl.load(x_ptr + offsets, mask=mask, other=0.0)
    else:
        x = tl.load(x_ptr + offsets, mask=mask, other=0.0)
    
    # Optimized softplus computation with numerical stability
    # Using more efficient mathematical formulations
    abs_x = tl.where(x > 0, x, -x)
    is_positive = x > 0
    
    # For x > 0: x + log1p(exp(-x))
    # For x <= 0: log1p(exp(x))
    # Using tl.exp with faster approximation for moderate values
    neg_x = -x
    exp_neg_x = tl.where(x < 10.0, tl.exp(neg_x), 0.0)  # exp(-10) ≈ 4.5e-5
    exp_x = tl.where(x > -10.0, tl.exp(x), 0.0)  # exp(-10) ≈ 4.5e-5
    
    # Compute log1p more efficiently
    log1p_exp_neg_x = tl.where(exp_neg_x < 1e-4, 
                               -x - 0.5 * exp_neg_x,  # Approximation for large positive x
                               tl.log(1.0 + exp_neg_x))
    
    log1p_exp_x = tl.where(exp_x < 1e-4,
                           x - 0.5 * exp_x,  # Approximation for large negative x
                           tl.log(1.0 + exp_x))
    
    y = tl.where(is_positive, x + log1p_exp_neg_x, log1p_exp_x)
    
    # Optimized tanh computation using rational approximation
    # tanh(y) = y * (27 + y^2) / (27 + 9 * y^2) for y in [-3, 3]

    y2 = y * y
    tanh_y = tl.where(abs_y < 3.0,
                     y * (27.0 + y2) / (27.0 + 9.0 * y2),
                     tl.where(y > 0, 1.0, -1.0))
    
    # Final output
    output = x * tanh_y
    
    # Vectorized store
    tl.store(output_ptr + offsets, output, mask=mask)
