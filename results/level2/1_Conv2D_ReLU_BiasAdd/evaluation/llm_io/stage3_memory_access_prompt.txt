import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.autotune(
    configs=[
        triton.Config({'BLOCK_SIZE_SPATIAL': 256}, num_warps=8),
        triton.Config({'BLOCK_SIZE_SPATIAL': 512}, num_warps=16),
        triton.Config({'BLOCK_SIZE_SPATIAL': 1024}, num_warps=32),
    ],
    key=['spatial_size'],
)
@triton.jit
def bias_add_kernel_2d(
    input_ptr,
    bias_ptr,
    output_ptr,
    batch_size,
    channels,
    spatial_size,
    BLOCK_SIZE_SPATIAL: tl.constexpr,
):
    """
    Optimized 2D kernel for bias addition.
    
    Grid dimensions:
    - pid_bc: flattened batch * channels index
    - pid_spatial: spatial block index
    
    Each thread block processes a contiguous spatial block for a specific
    (batch, channel) pair, enabling better memory coalescing.
    """
    pid_bc = tl.program_id(axis=0)
    pid_spatial = tl.program_id(axis=1)
    
    # Reconstruct batch and channel indices
    batch_idx = pid_bc // channels
    channel_idx = pid_bc % channels
    
    # Spatial indices for this block
    spatial_offsets = pid_spatial * BLOCK_SIZE_SPATIAL + tl.arange(0, BLOCK_SIZE_SPATIAL)
    spatial_mask = spatial_offsets < spatial_size
    
    # Base pointer for this (batch, channel) pair
    base_offset = (batch_idx * channels + channel_idx) * spatial_size
    input_block_ptr = input_ptr + base_offset
    output_block_ptr = output_ptr + base_offset
    
    # Load bias for this channel (single value, broadcast)
    bias = tl.load(bias_ptr + channel_idx)
    
    # Load input block, add bias, and store
    input_vals = tl.load(input_block_ptr + spatial_offsets, mask=spatial_mask, other=0.0)
    output_vals = input_vals + bias
    tl.store(output_block_ptr + spatial_offsets, output_vals, mask=spatial_mask)


def triton_bias_add_optimized(input_tensor: torch.Tensor, bias: torch.Tensor) -> torch.Tensor:
    """
    Optimized wrapper function for bias addition.
    
    Args:
        input_tensor: Shape [batch, channels, height, width]
        bias: Shape [channels, 1, 1]
    
    Returns:
        Tensor with same shape as input_tensor
    """
    # Ensure tensors are contiguous
    input_tensor = input_tensor.contiguous()
    bias = bias.contiguous()
    
    # Get dimensions
    batch_size, channels, height, width = input_tensor.shape
    spatial_size = height * width
    
    # Reshape bias to 1D
    bias_1d = bias.view(channels)
    
    # Allocate output
    output = torch.empty_like(input_tensor)
    
    # Grid configuration
    grid = lambda META: (
        batch_size * channels,  # flattened batch*channel dimension
        triton.cdiv(spatial_size, META['BLOCK_SIZE_SPATIAL']),
    )
    
    # Launch optimized kernel
    bias_add_kernel_2d[grid](
        input_tensor,
        bias_1d,
        output,
        batch_size,
        channels,
        spatial_size,
    )
    
    return output
