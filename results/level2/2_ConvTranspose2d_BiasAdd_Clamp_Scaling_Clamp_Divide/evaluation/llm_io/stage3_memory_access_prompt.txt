import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.autotune(
    configs=[
        triton.Config({'BLOCK_SIZE': 256}, num_warps=4, num_stages=3),
        triton.Config({'BLOCK_SIZE': 512}, num_warps=8, num_stages=3),
        triton.Config({'BLOCK_SIZE': 1024}, num_warps=8, num_stages=2),
        triton.Config({'BLOCK_SIZE': 2048}, num_warps=8, num_stages=1),
    ],
    key=['B', 'C', 'H', 'W']
)
@triton.jit
def post_process_kernel(
    x_ptr,
    bias_ptr,
    output_ptr,
    scaling_factor,
    B, C, H, W,
    BLOCK_SIZE: tl.constexpr,
):
    # 3D grid: (batch, channel, spatial_blocks)
    batch_id = tl.program_id(0)
    channel_id = tl.program_id(1)
    spatial_block_id = tl.program_id(2)
    
    spatial_size = H * W
    base_offset = batch_id * C * spatial_size + channel_id * spatial_size
    
    # Spatial processing with vectorization
    spatial_start = spatial_block_id * BLOCK_SIZE
    offsets = base_offset + spatial_start + tl.arange(0, BLOCK_SIZE)
    mask = spatial_start + tl.arange(0, BLOCK_SIZE) < spatial_size
    
    # Load input
    x = tl.load(x_ptr + offsets, mask=mask)
    
    # Load bias once per channel
    bias = tl.load(bias_ptr + channel_id)
    
    # Fused operations with vectorization
    x = x + bias
    x = tl.minimum(tl.maximum(x, 0.0), 1.0)
    x = x * scaling_factor
    x = tl.minimum(tl.maximum(x, 0.0), 1.0)
    x = x * (1.0 / scaling_factor)
    
    # Store result
    tl.store(output_ptr + offsets, x, mask=mask)

def triton_post_process(
    x: torch.Tensor,
    bias: torch.Tensor,
    scaling_factor: float
) -> torch.Tensor:
    # Ensure tensors are contiguous and on correct device
    assert x.is_contiguous(), "Input tensor must be contiguous"
    assert bias.is_contiguous(), "Bias tensor must be contiguous"
    
    # Reshape bias for broadcasting
    bias_reshaped = bias.view(-1)
    
    # Allocate output
    output = torch.empty_like(x)
    
    # Get tensor dimensions
    B, C, H, W = x.shape
    
    # 3D grid for better SM utilization
    spatial_size = H * W
    grid = lambda META: (B, C, triton.cdiv(spatial_size, META['BLOCK_SIZE']))
    
    # Launch kernel
    post_process_kernel[grid](
        x, bias_reshaped, output,
        scaling_factor,
        B, C, H, W,
    )
    
    return output
