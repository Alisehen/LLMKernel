```python
import torch
import torch.nn as nn
import triton
import triton.language as tl
import math


@triton.autotune(
    configs=[
        # Primary: Medium block size, high occupancy
        triton.Config({'BLOCK_M': 64, 'BLOCK_N': 64, 'BLOCK_K': 32}, num_stages=4, num_warps=8),
        # Conservative: Smaller blocks, lower register pressure
        triton.Config({'BLOCK_M': 32, 'BLOCK_N': 32, 'BLOCK_K': 32}, num_stages=5, num_warps=4),
        # Aggressive: Larger blocks for arithmetic intensity
        triton.Config({'BLOCK_M': 128, 'BLOCK_N': 64, 'BLOCK_K': 32}, num_stages=3, num_warps=8),
    ],
    key=['M', 'N', 'K'],
)
@triton.jit
def fused_gemm_mul_leaky_relu_kernel(
    # Pointers to matrices
    a_ptr, b_ptr, bias_ptr, c_ptr,
    # Matrix dimensions
    M, N, K,
    # Strides
    stride_am, stride_ak,
    stride_bk, stride_bn,
    stride_cm, stride_cn,
    # Scalar parameters
    multiplier,
    negative_slope,
    # Tile sizes
    BLOCK_M: tl.constexpr,
    BLOCK_N: tl.constexpr,
    BLOCK_K: tl.constexpr,
    # Use TF32 for tensor cores
    USE_TF32: tl.constexpr = True,
    # Enable software pipelining
    USE_SOFTWARE_PIPELINING: tl.constexpr = True,
):
    """
    Optimized fused kernel: GEMM → Bias Add → Multiply → LeakyReLU
    Computes: C = LeakyReLU((A @ B^T + bias) * multiplier)
    
    Optimizations:
    1. Software pipelining for memory/compute overlap
    2. Optimized block sizes for register pressure (64x64 main config)
    3. Fused LeakyReLU with cheap recomputation instead of storing intermediates
    """
    # -----------------------------------------------------------
    # Map program ids to block of C
    pid = tl.program_id(0)
    num_pid_n = tl.cdiv(N, BLOCK_N)
    pid_m = pid // num_pid_n
    pid_n = pid % num_pid_n
    
    # ----------------------------------------------------------
    # Create block pointers with offsets
    offs_am = pid_m * BLOCK_M + tl.arange(0, BLOCK_M)
    offs_bn = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)
    offs_k = tl.arange(0, BLOCK_K)
    
    # ----------------------------------------------------------
    # Initialize accumulator
    acc = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)
    
    # ----------------------------------------------------------
    # Initialize a and b to zeros for safety
    a = tl.zeros((BLOCK_M, BLOCK_K), dtype=tl.float32)
    b = tl.zeros((BLOCK_K, BLOCK_N), dtype=tl.float32)
    
    # ----------------------------------------------------------
    # Software pipelining: Load first blocks
    k = 0
    k_offset = k * BLOCK_K
    k_remaining = K - k_offset
    
    if USE_SOFTWARE_PIPELINING and k_remaining > 0:
        # Load first A block
        a_ptrs = a_ptr + (offs_am[:, None] * stride_am + 
                         (offs_k[None, :] + k_offset) * stride_ak)
        a_mask = (offs_am[:, None] < M) & (offs_k[None, :] < k_remaining)
        a = tl.load(a_ptrs, mask=a_mask, other=0.0)
        
        # Load first B block
        b_ptrs = b_ptr + ((offs_k[:, None] + k_offset) * stride_bk + 
                         offs_bn[None, :] * stride_bn)
        b_mask = (offs_k[:, None] < k_remaining) & (offs_bn[None, :] < N)
        b = tl.load(b_ptrs, mask=b_mask, other=0.0)
        
        # Pre-increment k for next iteration
        k += 1
        k_offset = k * BLOCK_K
        k_remaining = K - k_offset
    elif not USE_SOFTWARE_PIPELINING and k_remaining > 0:
        # Load first block for non-pipelined path
        a_ptrs = a_ptr + (offs_am[:, None] * stride_am + 
                         (offs_k[None, :] + k_offset) * stride_ak)
        a_mask = (offs_am[:, None] < M) & (offs_k[None, :] < k_remaining)
        a = tl.load(a_ptrs, mask=a_mask, other=0.0)
        
        b_ptrs = b_ptr + ((offs_k[:, None] + k_offset) * stride_bk + 
                         offs_bn[None, :] * stride_bn)
        b_mask = (offs_k[:, None] < k_remaining) & (offs_bn[None, :] < N)
        b = tl.load(b_ptrs, mask=b_mask, other=0.0)
    
    # ----------------------------------------------------------
    # Main computation loop
    while k < tl.cdiv(K, BLOCK_K):
        # Load next A and B blocks for software pipelining
        if USE_SOFTWARE_PIPELINING and k_remaining > 0:
            next_a_ptrs = a_ptr + (offs_am[:, None] * stride_am + 
                                 (offs_k[None, :] + k_offset) * stride_ak)
            next_a_mask = (offs_am[:, None] < M) & (offs_k[None, :] < k_remaining)
            next_a = tl.load(next_a_ptrs, mask=next_a_mask, other=0.0)
            
            next_b_ptrs = b_ptr + ((offs_k[:, None] + k_offset) * stride_bk + 
                                 offs_bn[None, :] * stride_bn)
            next_b_mask = (offs_k[:, None] < k_remaining) & (offs_bn[None, :] < N)
            next_b = tl.load(next_b_ptrs, mask=next_b_mask, other=0.0)
        
        # Compute dot product with current blocks
        acc += tl.dot(a, b, allow_tf32=USE_TF32)
        
        # Move next to current for next iteration (software pipelining)
        if USE_SOFTWARE_PIPELINING and k_remaining > 0:
            a = next_a
            b = next_b
        
        # Prepare for next iteration
        k += 1
        k_offset = k * BLOCK_K
        k_remaining = K - k_offset
        
        # Load next blocks for non-software-pipelining
        if not USE_SOFTWARE_PIPELINING and k_remaining > 0:
            a_ptrs = a_ptr + (offs_am[:, None] * stride_am + 
                             (offs_k[None, :] + k_offset) * stride_ak)
            a_mask = (offs_am[:, None] < M) & (offs_k[None, :] < k_remaining)
            a = tl.load(a_ptrs, mask=a_mask, other=0.0)
            
            b_ptrs = b_ptr + ((offs_k[:, None] + k_offset) * stride_bk + 
                             offs_bn[None, :] * stride_bn)
            b_mask = (offs_k[:, None] < k_remaining) & (offs_bn[None, :] < N)
            b = tl.load(b_ptrs, mask=b_mask, other=0.0)
    
    # ----------------------------------------------------------
    # Final dot product if software pipelining was used and we have remaining
    if USE_SOFTWARE_PIPELINING and K > 0 and (K % BLOCK_K != 0 or K <= BLOCK_K):
        acc += tl.dot(a, b, allow_tf32=USE_TF32)
    
    # ----------------------------------------------------------
    # Add bias (broadcast over rows)
    bias_ptrs = bias_ptr + offs_bn
    bias_mask = offs_bn < N
    bias = tl.load(bias_ptrs, mask=bias_mask, other=0.0)
    acc = acc + bias[None, :]  # Broadcast bias across rows
    
    # ----------------------------------------------------------
    # Fused Multiply → LeakyReLU with recomputation optimization
    # Apply multiplier
    acc = acc * multiplier
    
    # Optimized LeakyReLU: avoid storing intermediates to reduce register pressure
    # Compute condition once and reuse
    is_negative = acc < 0.0
    # Use conditional scaling instead of separate min/max
    acc = tl.where(is_negative, acc * negative_slope, acc)
    
    # ----------------------------------------------------------
    # Write back the block of C
    offs_cm = pid_m * BLOCK_M + tl.arange(0, BLOCK_M)
    offs_cn = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)
    
    c_ptrs = c_ptr + (offs_cm[:, None] * stride_cm + 
                      offs_cn[None, :] * stride_cn)
    c_mask = (offs_cm[:, None] < M) & (offs_cn[None, :] < N)
    tl.store(c_ptrs, acc, mask=c_mask)


def fused_gemm_mul_leaky_relu(
    x: torch.Tensor,
    weight: torch.Tensor,
    bias: torch.Tensor,
    multiplier: float,
    negative_slope: float,
) -> torch.Tensor:
    """
    Optimized fused GEMM + Bias Add + Multiply + LeakyReLU.
    
    Args:
        x: Input tensor of shape (M, K)
        weight: Weight tensor of shape (N, K)
        bias: Bias tensor of shape (N,)
        multiplier: Scalar multiplier
        negative_slope: Slope for negative values in LeakyReLU
    """
    # Check inputs
    assert x.dim() == 2, f"Expected 2D input, got {x.dim()}D"
    M, K = x.shape
    N = weight.shape[0]
    assert weight.shape[1] == K, f"Shape mismatch: x {x.shape}, weight {weight.shape}"
    assert bias.shape == (N,), f"Bias shape mismatch: expected ({N},), got {bias.shape}"
    
    # Prepare output
    c = torch.empty((M, N), device=x.device, dtype=x.dtype)
    
    # Ensure tensors are contiguous
    x = x.contiguous()
    weight_t = weight.t().contiguous()
    bias = bias.contiguous()
    
    # Grid calculation
    def grid(meta):
        return (triton.cdiv(M, meta['BLOCK_M']) * triton.cdiv(N, meta['BLOCK_N']),)
    
    # Launch autotuned kernel
    fused_gemm_mul_leaky_relu_kernel[grid](
        x, weight_t, bias, c,
        M, N, K,
        x.stride(0), x.stride(1),
        weight_t.stride(0), weight_t.stride(1),
        c.stride(0), c.stride(1),
        multiplier,
        negative_slope,
    )
    
    return c


class ModelNew(nn.Module):
    """
    Optimized model with fused GEMM + Bias Add + Multiply + LeakyReLU.
    Uses autotuned Triton kernels for maximum performance.
    """
    def __init__(self, in_features, out_features, multiplier, negative_slope):
        super(ModelNew, self).__init__()
        self.weight = nn.Parameter(torch.randn(out_features, in_features))
        self.bias = nn.Parameter(torch.randn(out_features))
        self.multiplier = multiplier
        self.negative_slope = negative_slope
        
        # Initialize parameters for stable training
        nn.init.kaiming_uniform_(self.weight, a=math.sqrt(5))
        fan_in, _ = nn.init._calculate_fan_in_and_fan_out(self.weight)
        bound = 1 / math.sqrt(fan_in) if fan_in > 0 else 0
        nn.init.uniform_(self.bias, -bound, bound)
        
    def forward(self, x):
        # Ensure input is contiguous for optimal memory access
        x = x.contiguous()
        
        # Use optimized fused kernel
        return fused_gemm_mul_leaky_relu(
            x, self.weight, self.bias, self.multiplier, self.negative_slope
        )
```
