import torch, torch.nn as nn, triton, triton.language as tl


@triton.jit
def mish2_kernel(
    x_ptr,
    n_elements,
    BLOCK_SIZE: tl.constexpr,
):
    """
    In-place fused Mish(Mish(x)) kernel.
    Operates over a flat 1D view of the tensor.
    """
    pid = tl.program_id(0)
    offsets = pid * BLOCK_SIZE + tl.arange(0, BLOCK_SIZE)
    mask = offsets < n_elements

    # Load original values
    x = tl.load(x_ptr + offsets, mask=mask, other=0.0)
    x_dtype = x.dtype
    x_fp = x.to(tl.float32)

    # ---------- Mish #1 ----------
    # Numerically stable softplus:
    # softplus(x) = log(1 + exp(x)) but with branching to avoid overflow
    pos = x_fp > 0.0
    sp_pos = x_fp + tl.log(1.0 + tl.exp(-x_fp))  # x + log(1 + exp(-x)) for x > 0
    sp_neg = tl.log(1.0 + tl.exp(x_fp))          # log(1 + exp(x)) for x <= 0
    sp = tl.where(pos, sp_pos, sp_neg)

    # tanh(softplus(x)) computed via sigmoid for numerical stability
    t = tl.sigmoid(2.0 * sp)
    tanh_sp = 2.0 * t - 1.0

    y_fp = x_fp * tanh_sp

    # ---------- Mish #2 ----------
    pos2 = y_fp > 0.0
    sp_pos2 = y_fp + tl.log(1.0 + tl.exp(-y_fp))
    sp_neg2 = tl.log(1.0 + tl.exp(y_fp))
    sp2 = tl.where(pos2, sp_pos2, sp_neg2)

    t2 = tl.sigmoid(2.0 * sp2)
    tanh_sp2 = 2.0 * t2 - 1.0

    y2_fp = y_fp * tanh_sp2

    # Cast back to original dtype and store in-place
    y2 = y2_fp.to(x_dtype)
    tl.store(x_ptr + offsets, y2, mask=mask)


def mish_mish_inplace(x: torch.Tensor) -> torch.Tensor:
    """
    Apply Mish(Mish(x)) to `x` in-place using a Triton kernel.
    Assumes `x` is on CUDA and contiguous.
    """
    assert x.is_cuda, "Triton kernel requires CUDA tensor"
    assert x.is_contiguous(), "Activation kernel currently assumes contiguous tensor"

    n_elements = x.numel()
    if n_elements == 0:
        return x

    BLOCK_SIZE = 1024  # power-of-two as required

    grid = lambda META: (
        triton.cdiv(max(n_elements, 1), META["BLOCK_SIZE"]),
    )

    mish2_kernel[grid](
        x,
        n_elements,
        BLOCK_SIZE=BLOCK_SIZE,
        num_warps=4,
        num_stages=2,
    )
    return x


def conv2d_mish2(x: torch.Tensor, weight: torch.Tensor, bias: torch.Tensor) -> torch.Tensor:
    """
    Hybrid implementation:
    - Use cuDNN via torch.nn.functional.conv2d for convolution
    - Apply fused Mish(Mish(.)) in a single Triton kernel pass
    """
    # Conv2d: stride=1, padding=0, dilation=1, groups=1 (to match reference)
    y = torch.nn.functional.conv2d(
        x, weight, bias,
        stride=1,
        padding=0,
        dilation=1,
        groups=1,
    )
    # Fused Mish(Mish(.)) in-place
    mish_mish_inplace(y)
    return y


class ModelNew(nn.Module):
    """
    Optimized model:
    - Convolution via cuDNN (torch.nn.functional.conv2d)
    - Fused Mish + Mish via a Triton pointwise kernel
    Behavior matches:
        y = conv2d(x)
        y = mish(y)
        y = mish(y)
    """
    def __init__(self, in_channels, out_channels, kernel_size):
        super(ModelNew, self).__init__()
        assert isinstance(kernel_size, int), "Only integer kernel_size is supported"
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kernel_size = kernel_size

        # Match the parameter interface of the previous custom conv:
        # weight: [Cout, Cin, Kh, Kw], bias: [Cout]
        self.weight = nn.Parameter(
            torch.randn(out_channels, in_channels, kernel_size, kernel_size)
        )
        self.bias = nn.Parameter(torch.randn(out_channels))

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return conv2d_mish2(x, self.weight, self.bias)
