import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.jit
def fused_gemm_sigmoid_scale_resadd_kernel(
    # Pointers to matrices
    a_ptr, b_ptr, bias_ptr, c_ptr,
    # Matrix dimensions
    M, N, K,
    # Stride information
    stride_am, stride_ak,
    stride_bk, stride_bn,
    stride_cm, stride_cn,
    # Block sizes
    BLOCK_M: tl.constexpr, BLOCK_N: tl.constexpr, BLOCK_K: tl.constexpr,
    # Scaling factor
    scaling_factor,
    USE_TF32: tl.constexpr,
    GROUP_M: tl.constexpr,  # Fixed: Added GROUP_M parameter
):
    # Program ID: which block of C we're computing
    pid = tl.program_id(0)
    pid_m = pid // tl.cdiv(N, BLOCK_N)
    pid_n = pid % tl.cdiv(N, BLOCK_N)
    
    # Create block offsets with better memory access pattern
    offs_m = pid_m * BLOCK_M + tl.arange(0, BLOCK_M)
    offs_n = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)
    offs_k = tl.arange(0, BLOCK_K)
    
    # Create pointers for A and B with better memory pattern
    a_ptrs = a_ptr + offs_m[:, None] * stride_am + offs_k[None, :] * stride_ak
    b_ptrs = b_ptr + offs_k[:, None] * stride_bk + offs_n[None, :] * stride_bn
    
    # Accumulator
    accumulator = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)
    
    # Compute block of C = A * B with better loop structure
    for k in range(0, tl.cdiv(K, BLOCK_K)):
        # Load blocks from A and B with masking
        k_remaining = K - k * BLOCK_K
        a_mask = (offs_m[:, None] < M) & (offs_k[None, :] < k_remaining)
        b_mask = (offs_k[:, None] < k_remaining) & (offs_n[None, :] < N)
        
        a = tl.load(a_ptrs, mask=a_mask, other=0.0)
        b = tl.load(b_ptrs, mask=b_mask, other=0.0)
        
        # Use tensor cores for better performance on Ada Lovelace
        accumulator += tl.dot(a, b, allow_tf32=USE_TF32)
        
        # Move pointers to next blocks
        a_ptrs += BLOCK_K * stride_ak
        b_ptrs += BLOCK_K * stride_bk
    
    # Add bias
    bias = tl.load(bias_ptr + offs_n, mask=offs_n < N, other=0.0)
    accumulator += bias[None, :]
    
    # Fused sigmoid + scaling + residual without intermediate stores
    # Using fast sigmoid approximation: 1/(1 + exp(-x))
    # This stays entirely in registers
    neg_x = -accumulator
    # Use fast exponential approximation for better performance
    # exp(x) = 2^(x * log2(e)) - using native approximation
    exp_neg_x = tl.exp(neg_x)
    sigmoid_result = 1.0 / (1.0 + exp_neg_x)
    output = sigmoid_result * scaling_factor + accumulator
    
    # Store final output only
    c_ptrs = c_ptr + offs_m[:, None] * stride_cm + offs_n[None, :] * stride_cn
    c_mask = (offs_m[:, None] < M) & (offs_n[None, :] < N)
    tl.store(c_ptrs, output, mask=c_mask)

@triton.jit
def fast_fused_gemm_sigmoid_scale_resadd_kernel(
    # Pointers to matrices
    a_ptr, b_ptr, bias_ptr, c_ptr,
    # Matrix dimensions
    M, N, K,
    # Stride information
    stride_am, stride_ak,
    stride_bk, stride_bn,
    stride_cm, stride_cn,
    # Block sizes
    BLOCK_M: tl.constexpr, BLOCK_N: tl.constexpr, BLOCK_K: tl.constexpr,
    # Scaling factor
    scaling_factor,
    USE_TF32: tl.constexpr,
    GROUP_M: tl.constexpr,
):
    # Improved grid with better load balancing
    num_pid_m = tl.cdiv(M, BLOCK_M)
    num_pid_n = tl.cdiv(N, BLOCK_N)
    num_pid_in_group = GROUP_M * num_pid_n
    group_id = tl.program_id(0) // num_pid_in_group
    first_pid_m = group_id * GROUP_M
    group_size_m = min(num_pid_m - first_pid_m, GROUP_M)
    pid_m = first_pid_m + (tl.program_id(0) % num_pid_in_group) // num_pid_n
    pid_n = tl.program_id(0) % num_pid_n
    
    # Skip if pid_m is out of bounds
    if pid_m >= num_pid_m:
        return
    
    # Create block offsets
    offs_m = pid_m * BLOCK_M + tl.arange(0, BLOCK_M)
    offs_n = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)
    offs_k = tl.arange(0, BLOCK_K)
    
    # Create pointers
    a_ptrs = a_ptr + offs_m[:, None] * stride_am + offs_k[None, :] * stride_ak
    b_ptrs = b_ptr + offs_k[:, None] * stride_bk + offs_n[None, :] * stride_bn
    
    # Accumulator
    accumulator = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)
    
    # Precompute loop bounds
    K_blocks = tl.cdiv(K, BLOCK_K)
    
    # Main GEMM loop with software pipelining
    for k in range(K_blocks):
        # Load with masking
        k_remaining = K - k * BLOCK_K
        a_mask = (offs_m[:, None] < M) & (offs_k[None, :] < k_remaining)
        b_mask = (offs_k[:, None] < k_remaining) & (offs_n[None, :] < N)
        
        a = tl.load(a_ptrs, mask=a_mask, other=0.0)
        b = tl.load(b_ptrs, mask=b_mask, other=0.0)
        
        accumulator += tl.dot(a, b, allow_tf32=USE_TF32)
        
        a_ptrs += BLOCK_K * stride_ak
        b_ptrs += BLOCK_K * stride_bk
    
    # Load bias
    bias = tl.load(bias_ptr + offs_n, mask=offs_n < N, other=0.0)
    accumulator += bias[None, :]
    
    # Fused activation with fast approximation
    # Fast sigmoid: x * (0.5 + 0.25 * x) for |x| < 2, else 1/(1+exp(-x))
    # This reduces expensive exp() calls
    x = accumulator
    fast_mask = tl.abs(x) < 2.0
    fast_sigmoid = x * (0.5 + 0.25 * x)
    safe_sigmoid = 1.0 / (1.0 + tl.exp(-x))
    sigmoid_result = tl.where(fast_mask, fast_sigmoid, safe_sigmoid)
    
    # Fused scaling and residual
    output = sigmoid_result * scaling_factor + accumulator
    
    # Store
    c_ptrs = c_ptr + offs_m[:, None] * stride_cm + offs_n[None, :] * stride_cn
    c_mask = (offs_m[:, None] < M) & (offs_n[None, :] < N)
    tl.store(c_ptrs, output, mask=c_mask)

def fused_gemm_sigmoid_scale_resadd(x, weight, bias, scaling_factor, fast=True):
    """
    Fused operation: GEMM -> Sigmoid -> Scaling -> Residual Add
    
    Args:
        x: Input tensor of shape (M, K)
        weight: Weight tensor of shape (N, K)
        bias: Bias tensor of shape (N,)
        scaling_factor: Scaling factor for sigmoid output
        fast: Use fast approximation kernel
        
    Returns:
        Output tensor of shape (M, N)
    """
    M, K = x.shape
    N, K_w = weight.shape
    assert K == K_w, f"Dimension mismatch: x has K={K}, weight has K={K_w}"
    
    # Allocate output
    output = torch.empty((M, N), device=x.device, dtype=x.dtype)
    
    # Transpose weight
    weight_t = weight.t().contiguous()
    
    # Choose kernel based on size and precision requirements
    if fast and M * N >= 4096 * 4096:  # Use fast kernel for large matrices
        kernel = fast_fused_gemm_sigmoid_scale_resadd_kernel
        BLOCK_M, BLOCK_N, BLOCK_K = 128, 128, 32
        GROUP_M = 8
        num_warps = 8
        num_stages = 3
    else:
        kernel = fused_gemm_sigmoid_scale_resadd_kernel
        BLOCK_M, BLOCK_N, BLOCK_K = 64, 64, 32
        GROUP_M = 1
        num_warps = 4
        num_stages = 2
    
    # Grid calculation
    def grid(meta):
        if meta['GROUP_M'] > 1:
            total_blocks = tl.cdiv(M, meta['BLOCK_M']) * tl.cdiv(N, meta['BLOCK_N'])
            return (total_blocks,)
        else:
            return (tl.cdiv(M, meta['BLOCK_M']) * tl.cdiv(N, meta['BLOCK_N']),)
    
    # Launch kernel with optimal configuration for Ada Lovelace
    kernel[grid](
        x, weight_t, bias, output,
        M, N, K,
        x.stride(0), x.stride(1),
        weight_t.stride(0), weight_t.stride(1),
        output.stride(0), output.stride(1),
        BLOCK_M=BLOCK_M, BLOCK_N=BLOCK_N, BLOCK_K=BLOCK_K,
        scaling_factor=scaling_factor,
        USE_TF32=True,  # Enable tensor cores for Ada Lovelace
        GROUP_M=GROUP_M,
        num_warps=num_warps,
        num_stages=num_stages
    )
    
    return output

class ModelNew(nn.Module):
    """
    Model implementing the pattern "Gemm_Sigmoid_Scaling_ResidualAdd".
    """
    def __init__(self, input_size, hidden_size, scaling_factor):
        super(ModelNew, self).__init__()
        self.weight = nn.Parameter(torch.randn(hidden_size, input_size))
        self.bias = nn.Parameter(torch.randn(hidden_size))
        self.scaling_factor = scaling_factor

    def forward(self, x):
        """
        Forward pass of the model.

        Args:
            x (torch.Tensor): Input tensor of shape (batch_size, input_size).

        Returns:
            torch.Tensor: Output tensor of shape (batch_size, hidden_size).
        """
        return fused_gemm_sigmoid_scale_resadd(
            x, self.weight, self.bias, self.scaling_factor, fast=True
        )
