import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.jit
def fused_gemm_sigmoid_scale_resadd_kernel_optimized(
    # Pointers to matrices
    a_ptr, b_ptr, bias_ptr, c_ptr,
    # Matrix dimensions
    M, N, K,
    # Stride information
    stride_am, stride_ak,
    stride_bk, stride_bn,
    stride_cm, stride_cn,
    # Block sizes
    BLOCK_M: tl.constexpr, BLOCK_N: tl.constexpr, BLOCK_K: tl.constexpr,
    # Group size for dot product
    GROUP_M: tl.constexpr,
    # Scaling factor
    scaling_factor,
    # Precision flags
    USE_FP16_ACC: tl.constexpr,
    USE_TF32: tl.constexpr,
):
    # --- 2D grid covering output space ---
    pid = tl.program_id(0)
    num_pid_n = tl.cdiv(N, BLOCK_N)
    num_pid_m = tl.cdiv(M, BLOCK_M)
    num_pid_in_group = GROUP_M * num_pid_n
    group_id = pid // num_pid_in_group
    first_pid_m = group_id * GROUP_M
    group_size_m = min(num_pid_m - first_pid_m, GROUP_M)
    pid_m = first_pid_m + (pid % group_size_m)
    pid_n = (pid % num_pid_in_group) // group_size_m
    
    # --- Shared offsets for ALL fused ops ---
    offs_m = pid_m * BLOCK_M + tl.arange(0, BLOCK_M)
    offs_n = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)
    offs_k = tl.arange(0, BLOCK_K)
    
    # --- Pointers for GEMM ---
    a_ptrs = a_ptr + (offs_m[:, None] * stride_am + offs_k[None, :] * stride_ak)
    b_ptrs = b_ptr + (offs_k[:, None] * stride_bk + offs_n[None, :] * stride_bn)
    
    # --- Mask for GEMM loads ---
    a_mask = (offs_m[:, None] < M) & (offs_k[None, :] < K)
    b_mask = (offs_k[:, None] < K) & (offs_n[None, :] < N)
    
    # --- Accumulator with optimal precision ---
    if USE_FP16_ACC:
        accumulator = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float16)
    else:
        accumulator = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)
    
    # --- Main GEMM loop ---
    for k in range(0, tl.cdiv(K, BLOCK_K)):
        a = tl.load(a_ptrs, mask=a_mask & (offs_k[None, :] < K - k * BLOCK_K), other=0.0)
        b = tl.load(b_ptrs, mask=b_mask & (offs_k[:, None] < K - k * BLOCK_K), other=0.0)
        
        # Use TF32 Tensor Cores when available
        accumulator += tl.dot(a, b, allow_tf32=USE_TF32)
        
        a_ptrs += BLOCK_K * stride_ak
        b_ptrs += BLOCK_K * stride_bk
    
    # --- Convert accumulator to FP32 for fused ops ---
    accumulator_f32 = accumulator.to(tl.float32)
    
    # --- Bias add (SAME offsets as GEMM) ---
    bias = tl.load(bias_ptr + offs_n, mask=offs_n < N, other=0.0)
    accumulator_f32 += bias[None, :]
    
    # --- Save original for residual ---
    original = accumulator_f32
    
    # --- Optimized sigmoid using tanh approximation (faster on Ada) ---
    # sigmoid(x) = 0.5 * tanh(0.5 * x) + 0.5
    half_x = accumulator_f32 * 0.5
    tanh_half_x = tl.tanh(half_x)  # Hardware-optimized on Ada
    sigmoid_result = tanh_half_x * 0.5 + 0.5
    
    # --- Scale and residual add ---
    output = sigmoid_result * scaling_factor + original
    
    # --- Store to output (SAME offsets/mask as GEMM) ---
    c_ptrs = c_ptr + (offs_m[:, None] * stride_cm + offs_n[None, :] * stride_cn)
    c_mask = (offs_m[:, None] < M) & (offs_n[None, :] < N)
    tl.store(c_ptrs, output, mask=c_mask)

def fused_gemm_sigmoid_scale_resadd_optimized(x, weight, bias, scaling_factor):
    """
    Optimized fused operation for Ada Lovelace.
    """
    M, K = x.shape
    N, K_w = weight.shape
    assert K == K_w, f"Dimension mismatch: x has K={K}, weight has K={K_w}"
    
    # Output tensor
    output = torch.empty((M, N), device=x.device, dtype=x.dtype)
    
    # Transpose weight for column-major access
    weight_t = weight.t().contiguous()
    
    # Auto-tune configuration for Ada 4090
    configs = [
        {'BLOCK_M': 128, 'BLOCK_N': 128, 'BLOCK_K': 32, 'GROUP_M': 8, 'USE_FP16_ACC': True, 'USE_TF32': False},
        {'BLOCK_M': 64, 'BLOCK_N': 64, 'BLOCK_K': 64, 'GROUP_M': 8, 'USE_FP16_ACC': True, 'USE_TF32': False},
        {'BLOCK_M': 128, 'BLOCK_N': 256, 'BLOCK_K': 32, 'GROUP_M': 8, 'USE_FP16_ACC': True, 'USE_TF32': False},
    ]
    
    def grid(meta):
        return (triton.cdiv(M, meta['BLOCK_M']) * triton.cdiv(N, meta['BLOCK_N']),)
    
    # Heuristic: Choose based on problem size
    if M >= 2048 and N >= 2048:
        config = configs[2]  # Large matrices
    elif K >= 512:
        config = configs[1]  # Deep matrices
    else:
        config = configs[0]  # Default
    
    # Launch kernel
    fused_gemm_sigmoid_scale_resadd_kernel_optimized[grid](
        x, weight_t, bias, output,
        M, N, K,
        x.stride(0), x.stride(1),
        weight_t.stride(0), weight_t.stride(1),
        output.stride(0), output.stride(1),
        BLOCK_M=config['BLOCK_M'],
        BLOCK_N=config['BLOCK_N'],
        BLOCK_K=config['BLOCK_K'],
        GROUP_M=config['GROUP_M'],
        scaling_factor=scaling_factor,
        USE_FP16_ACC=config['USE_FP16_ACC'],
        USE_TF32=config['USE_TF32'],
        num_warps=8 if config['BLOCK_M'] * config['BLOCK_N'] >= 16384 else 4,
        num_stages=3,
    )
    
    return output

class ModelNew(nn.Module):
    """
    Optimized model using fused kernel with grid optimization.
    """
    def __init__(self, input_size, hidden_size, scaling_factor):
        super(ModelNew, self).__init__()
        self.weight = nn.Parameter(torch.randn(hidden_size, input_size))
        self.bias = nn.Parameter(torch.randn(hidden_size))
        self.scaling_factor = scaling_factor

    def forward(self, x):
        return fused_gemm_sigmoid_scale_resadd_optimized(
            x, self.weight, self.bias, self.scaling_factor
        )
