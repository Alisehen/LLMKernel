import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.autotune(
    configs=[
        # Higher arithmetic intensity, moderate tile - good default
        triton.Config(
            {'BLOCK_M': 64, 'BLOCK_N': 32, 'BLOCK_K': 32},
            num_warps=4,
            num_stages=4,
        ),
        # Alternate aspect ratio for different shapes
        triton.Config(
            {'BLOCK_M': 32, 'BLOCK_N': 64, 'BLOCK_K': 32},
            num_warps=4,
            num_stages=4,
        ),
        # Conservative fallback to reduce register pressure even further
        triton.Config(
            {'BLOCK_M': 32, 'BLOCK_N': 32, 'BLOCK_K': 32},
            num_warps=2,
            num_stages=3,
        ),
    ],
    key=[
        'N', 'C_in', 'H', 'W',
        'C_out', 'KH', 'KW',
        'H_out', 'W_out',
    ],
)
@triton.jit
def conv2d_scale_min_kernel(
    x_ptr, w_ptr, b_ptr, out_ptr,
    N, C_in, H, W, C_out,
    KH, KW,
    H_out, W_out,
    stride_xn, stride_xc, stride_xh, stride_xw,
    stride_wn, stride_wc, stride_wkh, stride_wkw,
    stride_on, stride_oc, stride_oh, stride_ow,
    scale,
    BLOCK_M: tl.constexpr,   # tile over P = N * H_out * W_out
    BLOCK_N: tl.constexpr,   # tile over C_out (channels) per block iteration
    BLOCK_K: tl.constexpr,   # tile over K = C_in * KH * KW
):
    # 1D launch grid over output positions P = N * H_out * W_out
    pid_m = tl.program_id(0)

    P = N * H_out * W_out
    HW_out = H_out * W_out
    K = C_in * KH * KW
    HW_filt = KH * KW

    # Offsets along flattened P dimension
    offs_m = pid_m * BLOCK_M + tl.arange(0, BLOCK_M)
    mask_m = offs_m < P

    # Decode flattened P index -> (n, oh, ow)
    n_idx = offs_m // HW_out
    rem = offs_m % HW_out
    oh_idx = rem // W_out
    ow_idx = rem % W_out

    # Base offsets for input and output (independent of channel / filter index)
    base_inp_offsets = (
        n_idx * stride_xn
        + oh_idx * stride_xh
        + ow_idx * stride_xw
    )
    base_out_offsets = (
        n_idx * stride_on
        + oh_idx * stride_oh
        + ow_idx * stride_ow
    )

    # Running minimum over channel dimension for each position in BLOCK_M
    running_min = tl.full((BLOCK_M,), 1e30, dtype=tl.float32)

    # Loop over output channels in tiles of BLOCK_N, completely
    # within a single program so we can reduce to min across channels.
    for oc0 in range(0, C_out, BLOCK_N):
        offs_n = oc0 + tl.arange(0, BLOCK_N)
        mask_n = offs_n < C_out

        # Accumulator for current [P_tile, C_out_tile]
        acc = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)

        # GEMM over K dimension (C_in * KH * KW)
        for k0 in range(0, K, BLOCK_K):
            offs_k = k0 + tl.arange(0, BLOCK_K)
            mask_k = offs_k < K

            # Map flattened K index -> (ic, kh, kw)
            ic = offs_k // HW_filt
            rem_k = offs_k % HW_filt
            kh = rem_k // KW
            kw = rem_k % KW

            # ----- Load input tile A: shape [BLOCK_M, BLOCK_K] -----
            x_offsets = (
                base_inp_offsets[:, None]
                + ic[None, :] * stride_xc
                + kh[None, :] * stride_xh
                + kw[None, :] * stride_xw
            )
            x_ptrs = x_ptr + x_offsets
            mask_x = mask_m[:, None] & mask_k[None, :]
            x_vals = tl.load(x_ptrs, mask=mask_x, other=0.0)

            # ----- Load weight tile B: shape [BLOCK_K, BLOCK_N] -----
            w_base_n = offs_n * stride_wn
            w_offsets = (
                w_base_n[None, :]
                + ic[:, None] * stride_wc
                + kh[:, None] * stride_wkh
                + kw[:, None] * stride_wkw
            )
            w_ptrs = w_ptr + w_offsets
            mask_w = mask_k[:, None] & mask_n[None, :]
            w_vals = tl.load(w_ptrs, mask=mask_w, other=0.0)

            # Matrix multiply accumulate: [M,K] x [K,N] -> [M,N]
            acc += tl.dot(x_vals, w_vals)

        # Fuse bias add + scale
        bias = tl.load(b_ptr + offs_n, mask=mask_n, other=0.0)
        acc += bias[None, :]
        acc = acc * scale

        # Per-position min within this channel tile, then update global running_min.
        # Use a static unrolled loop over BLOCK_N so indexing stays inside Triton JIT.
        block_min = tl.full((BLOCK_M,), 1e30, dtype=tl.float32)

        for j in tl.static_range(BLOCK_N):
            col_valid = mask_n[j]
            col_vals = acc[:, j]
            # Only update for valid channels and valid P indices
            update_mask = col_valid & mask_m
            block_min = tl.where(update_mask, tl.minimum(block_min, col_vals), block_min)

        running_min = tl.minimum(running_min, block_min)

    # Store final per-position minima: shape [N, 1, H_out, W_out]
    out_ptrs = out_ptr + base_out_offsets
    tl.store(out_ptrs, running_min, mask=mask_m)


def conv_scale_min_triton(x, weight, bias, scale_factor):
    """
    Fused Triton implementation of:
        y = Conv2d(x)
        y = y * scale_factor
        out = min(y, dim=1, keepdim=True)
    Performs convolution, bias add, scaling, and channel-wise min
    reduction in a single kernel for maximum performance.
    """
    x = x.contiguous()
    weight = weight.contiguous()
    bias = bias.contiguous()

    N, C_in, H, W = x.shape
    C_out, C_in_w, KH, KW = weight.shape
    assert C_in == C_in_w, "Input channels mismatch between x and weight"

    # No padding, stride = 1
    H_out = H - KH + 1
    W_out = W - KW + 1

    out = torch.empty((N, 1, H_out, W_out), device=x.device, dtype=x.dtype)

    stride_xn, stride_xc, stride_xh, stride_xw = x.stride()
    stride_wn, stride_wc, stride_wkh, stride_wkw = weight.stride()
    stride_on, stride_oc, stride_oh, stride_ow = out.stride()

    P = N * H_out * W_out

    def grid(meta):
        return (triton.cdiv(P, meta['BLOCK_M']),)

    conv2d_scale_min_kernel[grid](
        x, weight, bias, out,
        N, C_in, H, W, C_out,
        KH, KW,
        H_out, W_out,
        stride_xn, stride_xc, stride_xh, stride_xw,
        stride_wn, stride_wc, stride_wkh, stride_wkw,
        stride_on, stride_oc, stride_oh, stride_ow,
        float(scale_factor),
    )
    return out


class ModelNew(nn.Module):
    """
    Triton implementation of:
        y = Conv2d(x)
        y = y * scale_factor
        y = min(y, dim=1, keepdim=True)
    Implemented as a single, aggressively optimized Triton kernel.
    """
    def __init__(self, in_channels, out_channels, kernel_size, scale_factor):
        super(ModelNew, self).__init__()
        if isinstance(kernel_size, int):
            kh = kw = kernel_size
        else:
            kh, kw = kernel_size
        self.weight = nn.Parameter(
            torch.randn(out_channels, in_channels, kh, kw)
        )
        self.bias = nn.Parameter(torch.randn(out_channels))
        self.scale_factor = float(scale_factor)

    def forward(self, x):
        # Expect x: [N, C_in, H, W] on CUDA
        return conv_scale_min_triton(x, self.weight, self.bias, self.scale_factor)
