import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def rowwise_scale_clamp_logsumexp_mish_kernel(
    x_ptr, out_ptr,
    M, N,
    stride_xm, stride_xn,
    stride_om,
    total_scale, clamp_min, clamp_max,
    BLOCK_M: tl.constexpr,
    BLOCK_N: tl.constexpr,
):
    """
    x_ptr:   (M, N) linear output (matmul + bias), row-major
    out_ptr: (M, 1) result: logsumexp over N, then x * mish(x)
    This kernel processes a BLOCK_M x N tile of rows at a time, performing
    a fused row-wise logsumexp + Mish + x*Mish(x) in float32.
    """

    pid_m = tl.program_id(0)

    # Offsets for rows processed by this program
    offs_m = pid_m * BLOCK_M + tl.arange(0, BLOCK_M)
    mask_m = offs_m < M  # [BLOCK_M]

    # Initialize running max and sum for streaming logsumexp, per row
    neg_inf = -1.0e30
    row_max = tl.full((BLOCK_M,), neg_inf, dtype=tl.float32)
    row_sum = tl.zeros((BLOCK_M,), dtype=tl.float32)

    # Main loop over columns in tiles of size BLOCK_N
    for start_n in range(0, N, BLOCK_N):
        offs_n = start_n + tl.arange(0, BLOCK_N)
        mask_n = offs_n < N  # [BLOCK_N]

        # 2D mask for the [BLOCK_M, BLOCK_N] tile
        mask = mask_m[:, None] & mask_n[None, :]  # [BLOCK_M, BLOCK_N]

        # Pointer arithmetic for the tile
        x_ptrs = x_ptr + offs_m[:, None] * stride_xm + offs_n[None, :] * stride_xn

        # Load tile and upcast to fp32
        x = tl.load(x_ptrs, mask=mask, other=0.0).to(tl.float32)  # [BLOCK_M, BLOCK_N]

        # Fused scaling (includes residual x + x -> factor 2) and clamp
        y = x * total_scale
        y = tl.maximum(y, clamp_min)
        y = tl.minimum(y, clamp_max)

        # For out-of-bounds columns/rows, set to -inf so they don't affect max/sum
        y = tl.where(mask, y, neg_inf)

        # Tile-wise max along N for each row
        tile_max = tl.max(y, axis=1)  # [BLOCK_M]

        # Streaming logsumexp update
        m_old = row_max
        s_old = row_sum
        m_new = tl.maximum(m_old, tile_max)  # [BLOCK_M]

        # Shift by new max, exponentiate, and sum along N
        y_shift = y - m_new[:, None]        # [BLOCK_M, BLOCK_N]
        exp_tile = tl.exp(y_shift)          # [BLOCK_M, BLOCK_N]
        sum_tile = tl.sum(exp_tile, axis=1) # [BLOCK_M]

        # Update running sum and max per row
        row_sum = s_old * tl.exp(m_old - m_new) + sum_tile
        row_max = m_new

    # Final logsumexp for each row: lse = log(sum(exp)) + max
    lse = tl.log(row_sum) + row_max  # [BLOCK_M], fp32

    # Mish activation:
    # mish(x) = x * tanh(softplus(x)), softplus(x) = log(1 + exp(x))
    sp = tl.log(1.0 + tl.exp(lse))
    t = tl.exp(2.0 * sp)
    tanh_sp = (t - 1.0) / (t + 1.0)
    mish_lse = lse * tanh_sp

    out_val = lse * mish_lse  # x * mish(x), [BLOCK_M]

    # Store result to (M, 1) with keepdim=True
    out_ptrs = out_ptr + offs_m * stride_om
    tl.store(out_ptrs, out_val, mask=mask_m)


def linear_scale_clamp_logsumexp_mish(x, weight, bias, scale_factor, clamp_min, clamp_max):
    """
    x:       (M, K)
    weight:  (N, K)  -- nn.Linear(out_features=N, in_features=K).weight
    bias:    (N,)
    returns: (M, 1)
    """
    assert x.is_cuda and weight.is_cuda and bias.is_cuda
    M, K = x.shape
    N = weight.shape[0]

    # Phase 1: use cuBLAS-backed GEMM via F.linear for matmul + bias
    y = torch.nn.functional.linear(x, weight, bias)  # (M, N), row-major contiguous

    # Output: logsumexp over dim=1 with keepdim=True
    out = torch.empty((M, 1), device=x.device, dtype=y.dtype)

    # Fold "x * scale_factor; x = x + x" into a single multiplier
    total_scale = float(scale_factor) * 2.0

    # Tuned tile sizes for RTX 4090
    BLOCK_M = 32
    BLOCK_N = 128

    grid = lambda META: (triton.cdiv(M, META['BLOCK_M']),)

    rowwise_scale_clamp_logsumexp_mish_kernel[grid](
        y, out,
        M, N,
        y.stride(0), y.stride(1),

        BLOCK_M=BLOCK_M,
        BLOCK_N=BLOCK_N,

    )
    return out
