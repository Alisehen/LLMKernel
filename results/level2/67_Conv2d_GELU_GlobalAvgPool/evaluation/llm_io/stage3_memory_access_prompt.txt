@triton.jit
def gelu_kernel(
    x_ptr,
    output_ptr,
    n_elements,
    BLOCK_SIZE: tl.constexpr,
):
    pid = tl.program_id(axis=0)
    block_start = pid * BLOCK_SIZE
    offsets = block_start + tl.arange(0, BLOCK_SIZE)
    mask = offsets < n_elements
    
    x = tl.load(x_ptr + offsets, mask=mask)
    
    # GELU approximation with reduced operations
    # 0.5 * x * (1 + tanh(0.7978845608028654 * (x + 0.044715 * x^3)))
    sqrt_2_over_pi = 0.7978845608028654
    x_cubed = x * x * x
    inner = x + 0.044715 * x_cubed
    z = sqrt_2_over_pi * inner
    # tanh(z) using fast approximation
    exp_2z = tl.exp(2.0 * z)
    tanh_z = (exp_2z - 1.0) / (exp_2z + 1.0)
    
    output = 0.5 * x * (1.0 + tanh_z)
    tl.store(output_ptr + offsets, output, mask=mask)

    n_elements = output.numel()
    # Optimized for Ada Lovelace: balance register usage and occupancy
