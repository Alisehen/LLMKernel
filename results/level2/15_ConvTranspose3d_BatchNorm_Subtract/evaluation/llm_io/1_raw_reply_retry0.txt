import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def bn3d_partial_reduce_kernel(
    x_ptr,               # float32* [N, C, D, H, W] contiguous => [N, C, S]
    partial_sum_ptr,     # float32* [C, NUM_TILES]
    partial_sq_ptr,      # float32* [C, NUM_TILES]
    N, C, S,             # int32
    T,                   # int32, T = N * S
    NUM_TILES,           # int32, number of tiles along T
    BLOCK_T: tl.constexpr,
):
    # 2D grid: [C, NUM_TILES]
    pid_c = tl.program_id(0)
    pid_tile = tl.program_id(1)

    # Tile over the flattened (N * S) dimension
    t_start = pid_tile * BLOCK_T
    offs_t = t_start + tl.arange(0, BLOCK_T)
    mask_t = offs_t < T

    # Map flattened index t -> (n, s)
    t = offs_t
    n = t // S
    s = t - n * S

    # x index: ((n * C) + pid_c) * S + s
    idx = ((n * C) + pid_c) * S + s

    x = tl.load(x_ptr + idx, mask=mask_t, other=0.0)
    x = x.to(tl.float32)

    # Compute partial sums over this tile
    sum_x = tl.sum(x, axis=0)
    sum_x2 = tl.sum(x * x, axis=0)

    # Store per-tile partial sums for this channel
    tile_offset = pid_c * NUM_TILES + pid_tile
    tl.store(partial_sum_ptr + tile_offset, sum_x)
    tl.store(partial_sq_ptr + tile_offset, sum_x2)


@triton.jit
def bn3d_final_reduce_kernel(
    partial_sum_ptr,     # float32* [C, NUM_TILES]
    partial_sq_ptr,      # float32* [C, NUM_TILES]
    mean_ptr,            # float32* [C]
    var_ptr,             # float32* [C]
    C,                   # int32
    NUM_TILES,           # int32
    T,                   # int32, T = N * S
    BLOCK_TILES: tl.constexpr,
):
    # One program per channel
    c = tl.program_id(0)

    offs = tl.arange(0, BLOCK_TILES)

    # Scalar accumulators to minimize register pressure
    sum_x = tl.zeros((), dtype=tl.float32)
    sum_x2 = tl.zeros((), dtype=tl.float32)

    # Reduce over all tiles for this channel
    for tile_start in range(0, NUM_TILES, BLOCK_TILES):
        tile_idx = tile_start + offs
        mask = tile_idx < NUM_TILES
        base = c * NUM_TILES + tile_idx

        part_sum = tl.load(partial_sum_ptr + base, mask=mask, other=0.0)
        part_sq = tl.load(partial_sq_ptr + base, mask=mask, other=0.0)

        # Reduce vector into scalar and accumulate
        sum_x += tl.sum(part_sum, axis=0)
        sum_x2 += tl.sum(part_sq, axis=0)

    E = T
    E_f = E.to(tl.float32)

    mean = sum_x / E_f
    var = sum_x2 / E_f - mean * mean

    tl.store(mean_ptr + c, mean)
    tl.store(var_ptr + c, var)


@triton.jit
def bn3d_norm_sub_mean_kernel(
    x_ptr,            # float32* [N, C, D, H, W] contiguous
    mean_ptr,         # float32* [C]
    var_ptr,          # float32* [C]
    weight_ptr,       # float32* [C]
    bias_ptr,         # float32* [C]
    out_ptr,          # float32* [N, C, D, H, W] contiguous
    N, C, S,          # int32 N, C, S = D*H*W
    eps,              # float32
    BLOCK_S: tl.constexpr,
):
    # One program per (n, c)
    nc = tl.program_id(0)
    n = nc // C
    c = nc - n * C

    mean_c = tl.load(mean_ptr + c)
    var_c = tl.load(var_ptr + c)
    gamma = tl.load(weight_ptr + c)
    beta = tl.load(bias_ptr + c)

    inv_std = 1.0 / tl.sqrt(var_c + eps)

    base = (n * C + c) * S

    # First pass: compute spatial mean after BN for this (n, c)
    sum_y = tl.zeros((), dtype=tl.float32)
    for s_start in range(0, S, BLOCK_S):
        offs = s_start + tl.arange(0, BLOCK_S)
        mask = offs < S

        # All fused ops (BN and accumulation) use the SAME offsets/mask
        x = tl.load(x_ptr + base + offs, mask=mask, other=0.0)
        x = x.to(tl.float32)
        y = ((x - mean_c) * inv_std) * gamma + beta

        # Reduce tile to scalar to keep register usage low
        sum_y += tl.sum(tl.where(mask, y, 0.0), axis=0)

    S_f = S.to(tl.float32)
    mean_spatial = sum_y / S_f

    # Second pass: write normalized output minus spatial mean
    for s_start in range(0, S, BLOCK_S):
        offs = s_start + tl.arange(0, BLOCK_S)
        mask = offs < S

        # Same offsets/mask for all fused ops (BN + subtract + store)
        x = tl.load(x_ptr + base + offs, mask=mask, other=0.0)
        x = x.to(tl.float32)
        y = ((x - mean_c) * inv_std) * gamma + beta
        y = y - mean_spatial
        tl.store(out_ptr + base + offs, y, mask=mask)


def bn3d_norm_sub_mean(x, weight, bias, eps):
    """
    BatchNorm3d (training semantics: per-batch channel-wise mean/var)
    followed by subtracting per-sample spatial mean, implemented in Triton.

    x: [N, C, D, H, W] contiguous, float32, CUDA
    weight, bias: [C]
    """
    assert x.is_cuda
    x = x.contiguous()
    N, C, D, H, W = x.shape
    S = D * H * W
    T = N * S  # Flattened (N, D, H, W) dimension per channel

    # Allocate per-channel statistics
    mean = torch.empty(C, device=x.device, dtype=torch.float32)
    var = torch.empty(C, device=x.device, dtype=torch.float32)

    # Parallel multi-block reduction over (N * S) for each channel
    BLOCK_T = 256
    num_tiles = triton.cdiv(T, BLOCK_T)

    partial_sum = torch.empty(
        (C * num_tiles,), device=x.device, dtype=torch.float32
    )
    partial_sq = torch.empty_like(partial_sum)

    # Kernel 1: compute per-channel partial sums over tiles of N*S
    grid_partial = lambda META: (C, num_tiles)
    bn3d_partial_reduce_kernel[grid_partial](
        x,
        partial_sum,
        partial_sq,
        N,
        C,
        S,
        T,
        num_tiles,
        BLOCK_T=BLOCK_T,
        num_warps=4,
    )

    # Kernel 2: reduce partial sums to final per-channel mean and variance
    BLOCK_TILES = 256
    grid_final = lambda META: (C,)
    bn3d_final_reduce_kernel[grid_final](
        partial_sum,
        partial_sq,
        mean,
        var,
        C,
        num_tiles,
        T,
        BLOCK_TILES=BLOCK_TILES,
        num_warps=1,
    )

    # Kernel 3: apply BN and subtract per-sample spatial mean
    out = torch.empty_like(x, dtype=torch.float32)
    grid_norm = lambda META: (N * C,)
    bn3d_norm_sub_mean_kernel[grid_norm](
        x,
        mean,
        var,
        weight,
        bias,
        out,
        N,
        C,
        S,
        eps,
        BLOCK_S=256,
        num_warps=4,
    )
    return out


class ModelNew(nn.Module):
    """
    ConvTranspose3d + BatchNorm3d (training-style per-batch stats) +
    subtraction of spatial mean, with BN + mean-sub fused in Triton.
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, bias=True):
        super(ModelNew, self).__init__()
        self.conv_transpose = nn.ConvTranspose3d(
            in_channels,
            out_channels,
            kernel_size,
            stride=stride,
            padding=padding,
            bias=bias,
        )
        # Keep BatchNorm3d for learnable parameters (weight, bias) and eps.
        self.batch_norm = nn.BatchNorm3d(out_channels)

    def forward(self, x):
        x = self.conv_transpose(x)
        # Use training-mode BatchNorm semantics: per-batch statistics
        weight = self.batch_norm.weight
        bias = self.batch_norm.bias
        eps = self.batch_norm.eps
        x = bn3d_norm_sub_mean(x, weight, bias, eps)
        return x
