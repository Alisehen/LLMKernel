def fused_linear_scale_residual_kernel(
    x_ptr,
    w_ptr,
    b_ptr,
    output_ptr,
    M, N, K,
    scaling_factor,
    stride_xm, stride_xk,
    stride_wn, stride_wk,

    # Allocate shared memory for tiles
    x_tile = tl.zeros((BLOCK_M, BLOCK_K), dtype=tl.float32)
    w_tile = tl.zeros((BLOCK_N, BLOCK_K), dtype=tl.float32)
