class Model(nn.Module):

    Model that performs a matrix multiplication, division, summation, and scaling.
    """

        super(Model, self).__init__()

        self.scaling_factor = scaling_factor

def fused_matmul_div_sum_scale(x: torch.Tensor, weight: torch.Tensor, scaling_factor: float) -> torch.Tensor:
    """
    Computes:
        y = scaling_factor * sum_{h} ( (x @ weight.T)[..., h] / 2 )
      which is algebraically equal to:
        y = (scaling_factor / 2) * x @ weight.sum(dim=0)
    Returns y with shape (batch_size, 1).
    """
    assert x.is_cuda and weight.is_cuda, "Inputs must be on CUDA device"
    assert x.dtype == weight.dtype, "x and weight must have the same dtype"

    M, K = x.shape
    H, K_w = weight.shape
    assert K_w == K, "Incompatible shapes between x and weight"

    # Precompute column-wise sum over hidden dimension
    weight_sum = weight.sum(dim=0).contiguous()

    alpha = float(scaling_factor) * 0.5

    y = torch.zeros((M, 1), device=x.device, dtype=x.dtype)

    device_index = x.device.index if x.device.index is not None else torch.cuda.current_device()
    sm_count = torch.cuda.get_device_properties(device_index).multi_processor_count

    # Heuristic: if 1D grid would underutilize SMs, use 2D kernel with K-splitting + atomics
    approx_block_m = 64
    grid_1d_est = (M + approx_block_m - 1) // approx_block_m
    use_2d = grid_1d_est < 4 * sm_count

    if use_2d:
        grid = lambda META: (
            triton.cdiv(M, META['BLOCK_M']),
            triton.cdiv(K, META['BLOCK_K']),
        )
        gemv_scaled_kernel_2d[grid](
            x,
            weight_sum,
            y.view(-1),
            M,
            K,
            x.stride(0),
            x.stride(1),
            y.stride(0),
            alpha,
        )
    else:
        grid = lambda META: (triton.cdiv(M, META['BLOCK_M']),)
        gemv_scaled_kernel_1d[grid](
            x,
            weight_sum,
            y.view(-1),
            M,
            K,
            x.stride(0),
            x.stride(1),
            y.stride(0),
            alpha,
        )

    return y
