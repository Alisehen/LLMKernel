import torch
import torch.nn as nn
import triton
import triton.language as tl
import math


@triton.jit
def fused_scale_pool_bias_scale_kernel(
    x_ptr, out_ptr,
    scale1_ptr, bias_ptr, scale2_ptr,
    N, C, D, H, W,
    stride_xn, stride_xc, stride_xd, stride_xh, stride_xw,
    stride_on, stride_oc, stride_od, stride_oh, stride_ow,
    POOL_K: tl.constexpr,
    BLOCK_C: tl.constexpr,
    BLOCK_DHW: tl.constexpr,
):
    """
    Fused kernel: Scale1 + AvgPool3d(2) + BiasAdd + Scale2
    Input: [N, C, D, H, W] -> Output: [N, C, D//2, H//2, W//2]
    POOL_K = 2 (pooling kernel size)
    """
    # Program indices
    pid_n = tl.program_id(0)
    pid_cdhw = tl.program_id(1)
    
    if pid_n >= N:
        return
    
    # Decode channel and spatial indices
    D_out = D // POOL_K
    H_out = H // POOL_K
    W_out = W // POOL_K
    DHW_out = D_out * H_out * W_out

    C_BLOCKS = tl.cdiv(C, BLOCK_C)
    DHW_BLOCKS = tl.cdiv(DHW_out, BLOCK_DHW)
    
    # Correctly compute pid_c_block and spatial block index
    total_blocks = C_BLOCKS * DHW_BLOCKS
    if pid_cdhw >= total_blocks:
        return
    
    pid_c_block = pid_cdhw // DHW_BLOCKS
    pid_spatial_block = pid_cdhw % DHW_BLOCKS
    
    # Channel offsets
    c_offs = pid_c_block * BLOCK_C + tl.arange(0, BLOCK_C)
    c_mask = c_offs < C
    
    # Load parameters
    scale1 = tl.load(scale1_ptr)
    scale2 = tl.load(scale2_ptr)
    bias_vals = tl.load(bias_ptr + c_offs, mask=c_mask, other=0.0)
    
    # Process multiple spatial positions within this block
    for local_spatial_idx in range(BLOCK_DHW):
        spatial_idx = pid_spatial_block * BLOCK_DHW + local_spatial_idx

        if spatial_idx < DHW_out:

            h_out = hw_out // W_out

            d_in_start = d_out * POOL_K
            h_in_start = h_out * POOL_K
            w_in_start = w_out * POOL_K

                d_in = d_in_start + dd
                for dh in range(POOL_K):
                    h_in = h_in_start + dh

                                   c_offs * stride_xc +

                                   h_in * stride_xh + 
                                   w_in * stride_xw)
                        
                        x_vals = tl.load(x_ptr + x_offset, mask=c_mask, other=0.0)
                        # Apply scale1 and accumulate
                        pool_sum += x_vals * scale1
            
            # Average pooling (divide by 8)

def fused_post_convtranspose(x, scale1, bias, scale2):
    """
    Fused: Scale1 + AvgPool3d(2) + BiasAdd + Scale2
    """
    N, C, D, H, W = x.shape
    D_out, H_out, W_out = D // 2, H // 2, W // 2
    out = torch.empty((N, C, D_out, H_out, W_out), 
                      device=x.device, dtype=x.dtype)
    
    # Configuration
    BLOCK_C = triton.next_power_of_2(C)
    # Limit BLOCK_C to avoid register pressure
    if BLOCK_C > 512:
        BLOCK_C = 512
    
    # Spatial block size - process multiple spatial positions per program
    MAX_SPATIAL_PER_BLOCK = 4
    DHW_out = D_out * H_out * W_out
    BLOCK_DHW = min(DHW_out, MAX_SPATIAL_PER_BLOCK)
    
    # Calculate grid
    C_BLOCKS = (C + BLOCK_C - 1) // BLOCK_C
    DHW_BLOCKS = (DHW_out + BLOCK_DHW - 1) // BLOCK_DHW
    grid = (N, C_BLOCKS * DHW_BLOCKS)
    
    # Launch kernel

        POOL_K=2,
        BLOCK_C=BLOCK_C,
        BLOCK_DHW=BLOCK_DHW,
    )
