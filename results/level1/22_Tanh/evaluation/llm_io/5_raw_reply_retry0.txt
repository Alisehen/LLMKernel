@triton.autotune(
    configs=[
        triton.Config({'BLOCK_SIZE': 128, 'VEC_SIZE': 2, 'NUM_STAGES': 1}, num_warps=4),
        triton.Config({'BLOCK_SIZE': 256, 'VEC_SIZE': 2, 'NUM_STAGES': 1}, num_warps=4),

        triton.Config({'BLOCK_SIZE': 512, 'VEC_SIZE': 4, 'NUM_STAGES': 1}, num_warps=8),
        triton.Config({'BLOCK_SIZE': 1024, 'VEC_SIZE': 4, 'NUM_STAGES': 1}, num_warps=8),
        triton.Config({'BLOCK_SIZE': 256, 'VEC_SIZE': 2, 'NUM_STAGES': 2}, num_warps=4),
        triton.Config({'BLOCK_SIZE': 512, 'VEC_SIZE': 2, 'NUM_STAGES': 2}, num_warps=8),
        triton.Config({'BLOCK_SIZE': 1024, 'VEC_SIZE': 2, 'NUM_STAGES': 2}, num_warps=8),
        triton.Config({'BLOCK_SIZE': 256, 'VEC_SIZE': 4, 'NUM_STAGES': 2}, num_warps=4),
        triton.Config({'BLOCK_SIZE': 512, 'VEC_SIZE': 4, 'NUM_STAGES': 2}, num_warps=8),
        triton.Config({'BLOCK_SIZE': 1024, 'VEC_SIZE': 4, 'NUM_STAGES': 2}, num_warps=8),
    ],
    key=['n_elements'],
)
@triton.jit
def tanh_kernel_optimized(
    x_ptr,
    output_ptr,
    n_elements,
    BLOCK_SIZE: tl.constexpr,
    VEC_SIZE: tl.constexpr,
    NUM_STAGES: tl.constexpr,
):
    pid = tl.program_id(axis=0)
    
    # Vectorized offsets - each thread processes VEC_SIZE elements
    block_start = pid * BLOCK_SIZE * VEC_SIZE
    offsets = block_start + tl.arange(0, BLOCK_SIZE)[:, None] * VEC_SIZE + tl.arange(0, VEC_SIZE)[None, :]
    offsets = tl.view(offsets, -1)
    
    mask = offsets < n_elements
    
    # Vectorized load
    x_vec = tl.load(x_ptr + offsets, mask=mask, other=0.0)
    
    # Optimized tanh computation with improved numerical stability
    # Use tanh(x) = 1 - 2/(exp(2x) + 1) for better precision
    exp_neg_2x = tl.exp(-2.0 * tl.abs(x_vec))
    tanh_result = tl.where(
        x_vec >= 0,
        1.0 - 2.0 / (1.0 + exp_neg_2x),
        2.0 / (1.0 + exp_neg_2x) - 1.0
    )
    
    # Vectorized store
    tl.store(output_ptr + offsets, tanh_result, mask=mask)


def triton_tanh_optimized(x: torch.Tensor) -> torch.Tensor:
    output = torch.empty_like(x)
    output_flat = output.view(-1)
    x_flat = x.view(-1)
    
    n_elements = x_flat.numel()
    
    # Calculate optimal grid size with vectorization
    grid = lambda meta: (triton.cdiv(n_elements, meta['BLOCK_SIZE'] * meta['VEC_SIZE']),)
    
    # Launch optimized kernel
    tanh_kernel_optimized[grid](
        x_flat, output_flat, n_elements,
        num_stages=2,  # Start with 2 for better latency hiding
    )
    
    return output


class ModelNew(nn.Module):
    def __init__(self):
        super(ModelNew, self).__init__()
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return triton_tanh_optimized(x)

## Key Optimizations Applied:

1. **Vectorized Memory Access (VEC_SIZE)**: Each thread processes 2 or 4 elements concurrently, improving memory throughput and reducing DRAM pressure.

2. **Autotuning Configuration**: Multiple configurations tested for optimal performance:
   - Varying BLOCK_SIZE (128-1024) and VEC_SIZE (2,4)
   - Different NUM_STAGES (1,2) for pipeline depth optimization
   - Optimized warp configurations (4 or 8 warps per block)

3. **Improved Tanh Computation**: 
   - Uses `tanh(x) = 1 - 2/(exp(2x) + 1)` for better numerical stability
   - Avoids potential overflow in `exp(2x)` for large inputs
   - Uses `tl.abs()` and conditional computation for symmetry

4. **Memory Access Pattern**:
   - Coalesced vector loads/stores with proper masking
   - Each thread block processes `BLOCK_SIZE * VEC_SIZE` elements
   - Grid calculation adapts to vectorization factor

5. **Pipeline Optimization**:
   - Starts with `num_stages=2` for better latency hiding
   - Autotuner explores different pipeline depths

The optimized kernel should significantly reduce `dram__throughput.avg.pct_of_peak_sustained_elapsed` (currently 92.94%) by improving memory access efficiency and increasing compute intensity through vectorization.
