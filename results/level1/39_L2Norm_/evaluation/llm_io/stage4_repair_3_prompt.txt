def triton_l2_norm(x: torch.Tensor, epsilon: float = 1e-8) -> torch.Tensor:
    assert x.is_cuda, "Input must be on CUDA device"
    assert x.dim() >= 2, "Input must have at least 2 dimensions"
    
    # Get input shape and strides
    batch_size, dim = x.shape[0], x.shape[1]
    
    # Use in-place computation to avoid OOM
    # Instead of creating a new output tensor, we'll compute norm and normalize in-place
    # First, compute norm sums
    norm_sums = torch.zeros(batch_size, device=x.device, dtype=x.dtype)
    
    # Ensure input is contiguous for optimal memory access
    x = x.contiguous()
    
    # Get strides
    stride_x_batch = x.stride(0) if x.dim() > 1 else 0
    stride_x_dim = x.stride(1) if x.dim() > 1 else x.stride(0)
    
    # Configure block sizes (tuned for A100)
    BLOCK_SIZE_BATCH = min(64, batch_size)  # Warp-aligned for better memory coalescing
    BLOCK_SIZE_DIM = min(256, dim)   # Good for cache line utilization
    
    # Calculate grid dimensions
    grid_batch = triton.cdiv(batch_size, BLOCK_SIZE_BATCH)
    grid_dim = triton.cdiv(dim, BLOCK_SIZE_DIM)
    
    # Launch first kernel to compute sums
    l2_norm_kernel[(
        grid_batch,
        grid_dim,
        1
    )](
        x,
        norm_sums,
        batch_size,
        dim,
        stride_x_batch,
        stride_x_dim,
        1,  # norm_sums stride
        BLOCK_SIZE_BATCH=BLOCK_SIZE_BATCH,
        BLOCK_SIZE_DIM=BLOCK_SIZE_DIM,
    )

    normalize_kernel[(
        grid_batch,
        grid_dim,
        1
    )](
        x,
        norm_sums,

        dim,
        epsilon,
        stride_x_batch,
        stride_x_dim,
        1,  # norm_sums stride
        stride_x_batch,  # Output stride same as input
        stride_x_dim,
        BLOCK_SIZE_BATCH=BLOCK_SIZE_BATCH,
        BLOCK_SIZE_DIM=BLOCK_SIZE_DIM,
    )
    
    return x
