import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.autotune(
    configs=[
        triton.Config({'BLOCK_SIZE_BATCH': 32, 'BLOCK_SIZE_DIM': 256}, num_warps=8),
        triton.Config({'BLOCK_SIZE_BATCH': 64, 'BLOCK_SIZE_DIM': 128}, num_warps=8),
        triton.Config({'BLOCK_SIZE_BATCH': 128, 'BLOCK_SIZE_DIM': 64}, num_warps=8),
        triton.Config({'BLOCK_SIZE_BATCH': 256, 'BLOCK_SIZE_DIM': 32}, num_warps=8),
    ],
    key=['batch_size', 'dim']
)
@triton.jit
def l2_norm_kernel(
    x_ptr,
    output_ptr,
    batch_size,
    dim,
    stride_x_batch,
    stride_x_dim,
    stride_out_batch,
    stride_out_dim,
    BLOCK_SIZE_BATCH: tl.constexpr,
    BLOCK_SIZE_DIM: tl.constexpr,
):
    # Optimized 2D block structure
    pid_batch = tl.program_id(0)
    pid_dim = tl.program_id(1)
    
    # Optimized offset calculation with prefetch hints
    batch_offsets = pid_batch * BLOCK_SIZE_BATCH + tl.arange(0, BLOCK_SIZE_BATCH)
    dim_offsets = pid_dim * BLOCK_SIZE_DIM + tl.arange(0, BLOCK_SIZE_DIM)
    
    # Create masks
    batch_mask = batch_offsets < batch_size
    dim_mask = dim_offsets < dim
    
    # Optimized memory access with prefetching
    x_ptrs = x_ptr + (
        batch_offsets[:, None] * stride_x_batch +
        dim_offsets[None, :] * stride_x_dim
    )
    
    # Load with cache hints
    x_block = tl.load(x_ptrs, 
                     mask=batch_mask[:, None] & dim_mask[None, :],
                     other=0.0,
                     cache_modifier=tl.CACHE_MODIFIER.L1_PREFETCH)
    
    # Fused computation with tile accumulation
    local_squares = x_block * x_block
    
    # Optimized reduction with tile accumulation
    # Tile dimension for better register reuse
    TILE_DIM = 8
    num_tiles = BLOCK_SIZE_DIM // TILE_DIM
    
    if tl.program_id(2) == 0:  # Reduction dimension
        acc = tl.zeros((BLOCK_SIZE_BATCH,), dtype=tl.float32)
        for tile_idx in range(num_tiles):
            tile_start = tile_idx * TILE_DIM
            tile_slice = local_squares[:, tile_start:tile_start + TILE_DIM]
            acc += tl.sum(tile_slice, axis=1)
        
        # Handle remainder
        remainder_start = num_tiles * TILE_DIM
        if remainder_start < BLOCK_SIZE_DIM:
            remainder_slice = local_squares[:, remainder_start:]
            acc += tl.sum(remainder_slice, axis=1)
        
        # Atomic add with cache hint
        sum_ptr = output_ptr + batch_offsets * stride_out_batch
        tl.atomic_add(sum_ptr, acc, mask=batch_mask)

@triton.autotune(
    configs=[
        triton.Config({'BLOCK_SIZE_BATCH': 32, 'BLOCK_SIZE_DIM': 256}, num_warps=8),
        triton.Config({'BLOCK_SIZE_BATCH': 64, 'BLOCK_SIZE_DIM': 128}, num_warps=8),
        triton.Config({'BLOCK_SIZE_BATCH': 128, 'BLOCK_SIZE_DIM': 64}, num_warps=8),
        triton.Config({'BLOCK_SIZE_BATCH': 256, 'BLOCK_SIZE_DIM': 32}, num_warps=8),
    ],
    key=['batch_size', 'dim']
)
@triton.jit
def normalize_kernel(
    x_ptr,
    norm_ptr,
    output_ptr,
    batch_size,
    dim,
    epsilon,
    stride_x_batch,
    stride_x_dim,
    stride_norm_batch,
    stride_out_batch,
    stride_out_dim,
    BLOCK_SIZE_BATCH: tl.constexpr,
    BLOCK_SIZE_DIM: tl.constexpr,
):
    # Optimized 2D block structure
    pid_batch = tl.program_id(0)
    pid_dim = tl.program_id(1)
    
    # Optimized offset calculation
    batch_offsets = pid_batch * BLOCK_SIZE_BATCH + tl.arange(0, BLOCK_SIZE_BATCH)
    dim_offsets = pid_dim * BLOCK_SIZE_DIM + tl.arange(0, BLOCK_SIZE_DIM)
    
    # Create masks
    batch_mask = batch_offsets < batch_size
    dim_mask = dim_offsets < dim
    
    # Load norm values with cache hint
    norm_ptrs = norm_ptr + batch_offsets * stride_norm_batch
    norm_values = tl.load(norm_ptrs, 
                         mask=batch_mask, 
                         other=0.0,
                         cache_modifier=tl.CACHE_MODIFIER.L1_PREFETCH)
    
    # Optimized sqrt and division
    norm_values = tl.sqrt(norm_values + epsilon)
    inv_norm = 1.0 / norm_values
    
    # Load input with prefetching
    x_ptrs = x_ptr + (
        batch_offsets[:, None] * stride_x_batch +
        dim_offsets[None, :] * stride_x_dim
    )
    x_block = tl.load(x_ptrs, 
                     mask=batch_mask[:, None] & dim_mask[None, :],
                     other=0.0,
                     cache_modifier=tl.CACHE_MODIFIER.L1_PREFETCH)
    
    # Normalize with tile-based computation
    TILE_DIM = 8
    num_tiles = BLOCK_SIZE_BATCH // TILE_DIM
    
    for tile_idx in range(num_tiles):
        tile_start = tile_idx * TILE_DIM
        tile_end = tile_start + TILE_DIM
        
        # Extract tile
        inv_norm_tile = inv_norm[tile_start:tile_end]
        x_tile = x_block[tile_start:tile_end, :]
        
        # Compute normalized tile
        normalized_tile = x_tile * inv_norm_tile[:, None]
        
        # Store tile
        out_ptrs = output_ptr + (
            batch_offsets[tile_start:tile_end, None] * stride_out_batch +
            dim_offsets[None, :] * stride_out_dim
        )
        tile_mask = batch_mask[tile_start:tile_end, None] & dim_mask[None, :]
        tl.store(out_ptrs, normalized_tile, mask=tile_mask)
    
    # Handle remainder
    remainder_start = num_tiles * TILE_DIM
    if remainder_start < BLOCK_SIZE_BATCH:
        inv_norm_remainder = inv_norm[remainder_start:]
        x_remainder = x_block[remainder_start:, :]
        normalized_remainder = x_remainder * inv_norm_remainder[:, None]
        
        out_ptrs = output_ptr + (
            batch_offsets[remainder_start:, None] * stride_out_batch +
            dim_offsets[None, :] * stride_out_dim
        )
        remainder_mask = batch_mask[remainder_start:, None] & dim_mask[None, :]
        tl.store(out_ptrs, normalized_remainder, mask=remainder_mask)

def triton_l2_norm(x: torch.Tensor, epsilon: float = 1e-8) -> torch.Tensor:
    assert x.is_cuda, "Input must be on CUDA device"
    assert x.dim() >= 2, "Input must have at least 2 dimensions"
    
    # Get input shape
    batch_size, dim = x.shape[0], x.shape[1]
    output = torch.empty_like(x)
    
    # Ensure input is contiguous
    x = x.contiguous()
    output = output.contiguous()
    
    # Get strides
    stride_x_batch = x.stride(0) if x.dim() > 1 else 0
    stride_x_dim = x.stride(1) if x.dim() > 1 else x.stride(0)
    
    # Allocate norm sums with proper alignment
    norm_sums = torch.zeros(batch_size, device=x.device, dtype=torch.float32)
    
    # Calculate grid dimensions - use 3D grid for better occupancy
    # We'll use the first autotune config as reference for grid calculation
    BLOCK_SIZE_BATCH = 64
    BLOCK_SIZE_DIM = 128
    
    grid_batch = triton.cdiv(batch_size, BLOCK_SIZE_BATCH)
    grid_dim = triton.cdiv(dim, BLOCK_SIZE_DIM)
    
    # Launch first kernel to compute sums
    # Use 3D grid with reduction dimension
    l2_norm_kernel[(
        grid_batch,
        grid_dim,
        2  # 3D grid for better occupancy
    )](
        x,
        norm_sums,
        batch_size,
        dim,
        stride_x_batch,
        stride_x_dim,
        1,  # norm_sums is 1D
        0,
        BLOCK_SIZE_BATCH=BLOCK_SIZE_BATCH,
        BLOCK_SIZE_DIM=BLOCK_SIZE_DIM,
    )
    
    # Launch second kernel for normalization
    normalize_kernel[(
        grid_batch,
        grid_dim,
        1
    )](
        x,
        norm_sums,
        output,
        batch_size,
        dim,
        epsilon,
        stride_x_batch,
        stride_x_dim,
        1,  # norm_sums stride
        output.stride(0) if output.dim() > 1 else 0,
        output.stride(1) if output.dim() > 1 else output.stride(0),
        BLOCK_SIZE_BATCH=BLOCK_SIZE_BATCH,
        BLOCK_SIZE_DIM=BLOCK_SIZE_DIM,
    )
    
    return output

class ModelNew(nn.Module):
    """
    L2 normalization layer with optimized Triton kernels.
    """
    def __init__(self, epsilon: float = 1e-8):
        """
        Initializes the L2Norm layer.

        Args:
            epsilon (float): Small value to avoid division by zero.
        """
        super(ModelNew, self).__init__()
        self.epsilon = epsilon

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Applies L2 normalization to the input tensor.

        Args:
            x (torch.Tensor): Input tensor of shape (*, dim, *).

        Returns:
            torch.Tensor: Output tensor with L2 normalization applied, same shape as input.
        """
        return triton_l2_norm(x, self.epsilon)
