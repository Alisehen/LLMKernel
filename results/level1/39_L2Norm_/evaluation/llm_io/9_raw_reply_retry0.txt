import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.autotune(
    configs=[
        triton.Config({'BLOCK_B': 128, 'BLOCK_D': 256, 'NUM_STAGES': 1}, num_warps=8),
        triton.Config({'BLOCK_B': 64, 'BLOCK_D': 512, 'NUM_STAGES': 2}, num_warps=8),
        triton.Config({'BLOCK_B': 32, 'BLOCK_D': 1024, 'NUM_STAGES': 3}, num_warps=8),
        triton.Config({'BLOCK_B': 128, 'BLOCK_D': 512, 'NUM_STAGES': 2}, num_warps=8),
        triton.Config({'BLOCK_B': 64, 'BLOCK_D': 256, 'NUM_STAGES': 1}, num_warps=4),
    ],
    key=['batch_size', 'dim']
)
@triton.jit
def l2_norm_kernel(
    x_ptr,
    output_ptr,
    batch_size,
    dim,
    stride_x_batch,
    stride_x_dim,
    stride_out_batch,
    stride_out_dim,
    BLOCK_B: tl.constexpr,
    BLOCK_D: tl.constexpr,
    NUM_STAGES: tl.constexpr,
):
    pid_batch = tl.program_id(0)
    pid_dim = tl.program_id(1)
    
    batch_start = pid_batch * BLOCK_B
    dim_start = pid_dim * BLOCK_D
    
    batch_offsets = batch_start + tl.arange(0, BLOCK_B)
    dim_offsets = dim_start + tl.arange(0, BLOCK_D)
    
    batch_mask = batch_offsets < batch_size
    dim_mask = dim_offsets < dim
    
    x_ptrs = x_ptr + (
        batch_offsets[:, None] * stride_x_batch +
        dim_offsets[None, :] * stride_x_dim
    )
    
    # Use float32 for accumulation regardless of input type
    local_sum = tl.zeros((BLOCK_B,), dtype=tl.float32)
    
    # Process in stages for better memory latency hiding
    for d in range(0, BLOCK_D, 32):
        d_offsets = dim_start + d + tl.arange(0, 32)
        d_mask = d_offsets < dim
        
        if tl.program_id(1) == 0:
            x_chunk = tl.load(x_ptrs, mask=batch_mask[:, None] & d_mask[None, :], other=0.0)
            local_sum += tl.sum(x_chunk * x_chunk, axis=1)
        x_ptrs += 32 * stride_x_dim
    
    # Atomic add with proper masking
    if tl.sum(batch_mask) > 0:
        out_ptrs = output_ptr + batch_offsets * stride_out_batch
        tl.atomic_add(out_ptrs, local_sum, mask=batch_mask)

@triton.autotune(
    configs=[
        triton.Config({'BLOCK_B': 128, 'BLOCK_D': 256, 'NUM_STAGES': 1}, num_warps=8),
        triton.Config({'BLOCK_B': 64, 'BLOCK_D': 512, 'NUM_STAGES': 2}, num_warps=8),
        triton.Config({'BLOCK_B': 32, 'BLOCK_D': 1024, 'NUM_STAGES': 3}, num_warps=8),
        triton.Config({'BLOCK_B': 128, 'BLOCK_D': 512, 'NUM_STAGES': 2}, num_warps=8),
        triton.Config({'BLOCK_B': 64, 'BLOCK_D': 256, 'NUM_STAGES': 1}, num_warps=4),
    ],
    key=['batch_size', 'dim']
)
@triton.jit
def normalize_kernel(
    x_ptr,
    norm_ptr,
    output_ptr,
    batch_size,
    dim,
    epsilon,
    stride_x_batch,
    stride_x_dim,
    stride_norm_batch,
    stride_out_batch,
    stride_out_dim,
    BLOCK_B: tl.constexpr,
    BLOCK_D: tl.constexpr,
    NUM_STAGES: tl.constexpr,
):
    pid_batch = tl.program_id(0)
    pid_dim = tl.program_id(1)
    
    batch_start = pid_batch * BLOCK_B
    dim_start = pid_dim * BLOCK_D
    
    batch_offsets = batch_start + tl.arange(0, BLOCK_B)
    dim_offsets = dim_start + tl.arange(0, BLOCK_D)
    
    batch_mask = batch_offsets < batch_size
    dim_mask = dim_offsets < dim
    
    # Load norm values with cache hint
    norm_ptrs = norm_ptr + batch_offsets * stride_norm_batch
    norm_values = tl.load(norm_ptrs, mask=batch_mask, cache_modifier='.cg')
    
    # Compute inverse norm
    inv_norm = 1.0 / (tl.sqrt(norm_values) + epsilon)
    
    # Pointer arithmetic for input and output
    x_ptrs = x_ptr + (
        batch_offsets[:, None] * stride_x_batch +
        dim_offsets[None, :] * stride_x_dim
    )
    out_ptrs = output_ptr + (
        batch_offsets[:, None] * stride_out_batch +
        dim_offsets[None, :] * stride_out_dim
    )
    
    # Process in vectorized chunks for better memory coalescing
    for d in range(0, BLOCK_D, 64):
        d_offsets = dim_start + d + tl.arange(0, 64)
        d_mask = d_offsets < dim
        
        if tl.sum(d_mask) > 0:
            x_chunk = tl.load(x_ptrs, mask=batch_mask[:, None] & d_mask[None, :], other=0.0)
            normalized = x_chunk * inv_norm[:, None]
            tl.store(out_ptrs, normalized, mask=batch_mask[:, None] & d_mask[None, :])
        
        x_ptrs += 64 * stride_x_dim
        out_ptrs += 64 * stride_out_dim

def triton_l2_norm(x: torch.Tensor, epsilon: float = 1e-8) -> torch.Tensor:
    assert x.is_cuda, "Input must be on CUDA device"
    assert x.dim() >= 2, "Input must have at least 2 dimensions"
    
    batch_size, dim = x.shape[0], x.shape[1]
    output = torch.empty_like(x)
    
    # Ensure contiguous memory layout
    if not x.is_contiguous():
        x = x.contiguous()
    output = output.contiguous()
    
    stride_x_batch = x.stride(0)
    stride_x_dim = x.stride(1) if x.dim() > 1 else 1
    stride_out_batch = output.stride(0)
    stride_out_dim = output.stride(1) if output.dim() > 1 else 1
    
    # Allocate norm tensor with proper dtype for atomic accumulation
    norm_sums = torch.zeros(batch_size, device=x.device, dtype=torch.float32)
    
    # Calculate grid dimensions
    grid_batch = triton.cdiv(batch_size, 128)  # Use base block size for grid calculation
    grid_dim = triton.cdiv(dim, 256)
    
    # Launch kernels with optimized configurations
    l2_norm_kernel[(
        grid_batch,
        grid_dim,
        1
    )](
        x,
        norm_sums,
        batch_size,
        dim,
        stride_x_batch,
        stride_x_dim,
        1,
        0,
    )
    
    normalize_kernel[(
        grid_batch,
        grid_dim,
        1
    )](
        x,
        norm_sums,
        output,
        batch_size,
        dim,
        epsilon,
        stride_x_batch,
        stride_x_dim,
        1,
        stride_out_batch,
        stride_out_dim,
    )
    
    return output

class ModelNew(nn.Module):
    """
    L2 normalization layer with optimized Triton kernels.
    """
    def __init__(self, epsilon: float = 1e-8):
        """
        Initializes the L2Norm layer.

        Args:
            epsilon (float): Small value to avoid division by zero.
        """
        super(ModelNew, self).__init__()
        self.epsilon = epsilon

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Applies L2 normalization to the input tensor.

        Args:
            x (torch.Tensor): Input tensor of shape (*, dim, *).

        Returns:
            torch.Tensor: Output tensor with L2 normalization applied, same shape as input.
        """
        return triton_l2_norm(x, self.epsilon)
