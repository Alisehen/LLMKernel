import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.jit
def l2_norm_reduce_kernel(
    x_ptr,
    norms_ptr,
    batch_size,
    dim,
    stride_x_batch,
    stride_x_dim,
    BLOCK_SIZE_BATCH: tl.constexpr,
    BLOCK_SIZE_DIM: tl.constexpr,
):
    """First kernel: compute sum of squares and reduce to norms"""
    pid_batch = tl.program_id(0)
    pid_dim = tl.program_id(1)
    
    # Compute offsets
    batch_offsets = pid_batch * BLOCK_SIZE_BATCH + tl.arange(0, BLOCK_SIZE_BATCH)
    dim_offsets = pid_dim * BLOCK_SIZE_DIM + tl.arange(0, BLOCK_SIZE_DIM)
    
    # Create masks
    batch_mask = batch_offsets < batch_size
    dim_mask = dim_offsets < dim
    
    # Load block
    x_ptrs = x_ptr + (
        batch_offsets[:, None] * stride_x_batch +
        dim_offsets[None, :] * stride_x_dim
    )
    
    x_block = tl.load(x_ptrs, 
                     mask=batch_mask[:, None] & dim_mask[None, :],
                     other=0.0)
    
    # Compute local sum of squares
    local_squares = x_block * x_block
    block_sum = tl.sum(local_squares, axis=1)
    
    # Atomic add to global norms buffer (mask ensures only valid threads write)
    norms_ptrs = norms_ptr + batch_offsets
    tl.atomic_add(norms_ptrs, block_sum, mask=batch_mask)

@triton.jit
def l2_norm_normalize_kernel(
    x_ptr,
    norms_ptr,
    output_ptr,
    batch_size,
    dim,
    stride_x_batch,
    stride_x_dim,
    stride_out_batch,
    stride_out_dim,
    BLOCK_SIZE_BATCH: tl.constexpr,
    BLOCK_SIZE_DIM: tl.constexpr,
    epsilon: tl.constexpr,
):
    """Second kernel: normalize using precomputed norms"""
    pid_batch = tl.program_id(0)
    pid_dim = tl.program_id(1)
    
    # Compute offsets
    batch_offsets = pid_batch * BLOCK_SIZE_BATCH + tl.arange(0, BLOCK_SIZE_BATCH)
    dim_offsets = pid_dim * BLOCK_SIZE_DIM + tl.arange(0, BLOCK_SIZE_DIM)
    
    # Create masks
    batch_mask = batch_offsets < batch_size
    dim_mask = dim_offsets < dim
    
    # Load norms for this batch block
    norms_ptrs = norms_ptr + batch_offsets
    norms = tl.load(norms_ptrs, mask=batch_mask, other=1.0)
    
    # Compute inverse norm with epsilon (safe division)
    inv_norm = tl.where(norms > 0.0, 1.0 / tl.sqrt(norms + epsilon), 0.0)
    
    # Load input block
    x_ptrs = x_ptr + (
        batch_offsets[:, None] * stride_x_batch +
        dim_offsets[None, :] * stride_x_dim
    )
    
    x_block = tl.load(x_ptrs,
                     mask=batch_mask[:, None] & dim_mask[None, :],
                     other=0.0)
    
    # Normalize
    normalized = x_block * inv_norm[:, None]
    
    # Store result
    out_ptrs = output_ptr + (
        batch_offsets[:, None] * stride_out_batch +
        dim_offsets[None, :] * stride_out_dim
    )
    
    tl.store(out_ptrs, normalized,
            mask=batch_mask[:, None] & dim_mask[None, :])

def triton_l2_norm(x: torch.Tensor, epsilon: float = 1e-8) -> torch.Tensor:
    """Wrapper function for L2 normalization using Triton kernels"""
    assert x.is_cuda, "Input must be on CUDA device"
    assert x.dim() >= 2, "Input must have at least 2 dimensions"
    
    # Get input shape
    batch_size, dim = x.shape[0], x.shape[1]
    output = torch.empty_like(x)
    
    # Ensure input is contiguous
    x = x.contiguous()
    output = output.contiguous()
    
    # Get strides
    stride_x_batch = x.stride(0)
    stride_x_dim = x.stride(1) if x.dim() > 1 else 1
    stride_out_batch = output.stride(0)
    stride_out_dim = output.stride(1) if output.dim() > 1 else 1
    
    # Configuration - tuned for performance
    BLOCK_SIZE_BATCH = 128
    BLOCK_SIZE_DIM = 256
    
    # Calculate grid dimensions
    grid_batch = triton.cdiv(batch_size, BLOCK_SIZE_BATCH)
    grid_dim = triton.cdiv(dim, BLOCK_SIZE_DIM)
    
    # Allocate norms buffer
    norms = torch.zeros(batch_size, device=x.device, dtype=torch.float32)
    
    # Launch reduction kernel
    l2_norm_reduce_kernel[(
        grid_batch,
        grid_dim,
    )](
        x,
        norms,
        batch_size,
        dim,
        stride_x_batch,
        stride_x_dim,
        BLOCK_SIZE_BATCH=BLOCK_SIZE_BATCH,
        BLOCK_SIZE_DIM=BLOCK_SIZE_DIM,
    )
    
    # Launch normalization kernel
    l2_norm_normalize_kernel[(
        grid_batch,
        grid_dim,
    )](
        x,
        norms,
        output,
        batch_size,
        dim,
        stride_x_batch,
        stride_x_dim,
        stride_out_batch,
        stride_out_dim,
        BLOCK_SIZE_BATCH=BLOCK_SIZE_BATCH,
        BLOCK_SIZE_DIM=BLOCK_SIZE_DIM,
        epsilon=epsilon,
    )
    
    return output

class ModelNew(nn.Module):
    """L2 normalization layer with optimized Triton kernel"""
    def __init__(self, epsilon: float = 1e-8):
        super(ModelNew, self).__init__()
        self.epsilon = epsilon

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return triton_l2_norm(x, self.epsilon)
