import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.autotune(
    configs=[
        # Original configs - preserved for stability
        triton.Config({'BLOCK_M': 128, 'BLOCK_N': 128, 'BLOCK_K': 32}, num_stages=3, num_warps=8),
        triton.Config({'BLOCK_M': 64, 'BLOCK_N': 64, 'BLOCK_K': 32}, num_stages=4, num_warps=4),
        triton.Config({'BLOCK_M': 32, 'BLOCK_N': 32, 'BLOCK_K': 32}, num_stages=5, num_warps=2),
        triton.Config({'BLOCK_M': 64, 'BLOCK_N': 128, 'BLOCK_K': 32}, num_stages=3, num_warps=8),
        triton.Config({'BLOCK_M': 128, 'BLOCK_N': 64, 'BLOCK_K': 32}, num_stages=3, num_warps=8),
        
        # Micro-optimizations: Increased warps for better occupancy
        triton.Config({'BLOCK_M': 128, 'BLOCK_N': 128, 'BLOCK_K': 32}, num_stages=2, num_warps=16),  # More warps for better occupancy
        triton.Config({'BLOCK_M': 128, 'BLOCK_N': 128, 'BLOCK_K': 32}, num_stages=4, num_warps=16),  # More warps, different stages
        triton.Config({'BLOCK_M': 64, 'BLOCK_N': 64, 'BLOCK_K': 32}, num_stages=3, num_warps=8),    # Increased warps from 4
        triton.Config({'BLOCK_M': 64, 'BLOCK_N': 64, 'BLOCK_K': 32}, num_stages=5, num_warps=8),    # Increased warps
        triton.Config({'BLOCK_M': 32, 'BLOCK_N': 32, 'BLOCK_K': 32}, num_stages=4, num_warps=4),    # Increased warps from 2
        triton.Config({'BLOCK_M': 64, 'BLOCK_N': 128, 'BLOCK_K': 32}, num_stages=2, num_warps=16),  # More warps
        triton.Config({'BLOCK_M': 128, 'BLOCK_N': 64, 'BLOCK_K': 32}, num_stages=4, num_warps=16),  # More warps
    ],
    key=['M', 'N', 'K', 'L']
)
@triton.jit
def batched_matmul_kernel(
    A_ptr,
    B_ptr,
    C_ptr,
    N, M, K, L,
    stride_An, stride_Am, stride_Ak,
    stride_Bk, stride_Bl,
    stride_Cn, stride_Cm, stride_Cl,
    BLOCK_M: tl.constexpr,
    BLOCK_N: tl.constexpr,
    BLOCK_K: tl.constexpr,
    USE_TENSOR_CORES: tl.constexpr,
    ACC_TYPE: tl.constexpr,
):
    """Optimized batched matmul kernel with 3D grid for maximum occupancy"""
    pid_batch = tl.program_id(0)
    pid_m = tl.program_id(1)
    pid_n = tl.program_id(2)
    
    # Offsets for the current block
    offs_m = pid_m * BLOCK_M + tl.arange(0, BLOCK_M)
    offs_n = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)
    offs_k = tl.arange(0, BLOCK_K)
    
    # Initialize accumulator
    acc = tl.zeros((BLOCK_M, BLOCK_N), dtype=ACC_TYPE)
    
    # Pointers to the start of the current batch
    A_batch_ptr = A_ptr + pid_batch * stride_An
    B_ptr_start = B_ptr  # B is 2D, no batch dimension
    C_batch_ptr = C_ptr + pid_batch * stride_Cn
    
    # Pre-compute as much as possible outside the loop
    a_ptrs_base = A_batch_ptr + offs_m[:, None] * stride_Am
    b_ptrs_base = B_ptr_start + offs_n[None, :] * stride_Bl
    
    # Loop over K dimension
    for k in range(0, tl.cdiv(K, BLOCK_K)):
        k_offset = k * BLOCK_K
        
        # Compute A pointer
        A_block_ptr = a_ptrs_base + (k_offset + offs_k[None, :]) * stride_Ak
        
        # Compute B pointer  
        B_block_ptr = b_ptrs_base + (k_offset + offs_k[:, None]) * stride_Bk
        
        # Load A and B with masking for boundaries
        a = tl.load(
            A_block_ptr,
            mask=(offs_m[:, None] < M) & ((k_offset + offs_k[None, :]) < K),
            other=0.0
        )
        b = tl.load(
            B_block_ptr,
            mask=((k_offset + offs_k[:, None]) < K) & (offs_n[None, :] < L),
            other=0.0
        )
        
        # Compute matrix multiplication
        if USE_TENSOR_CORES and ACC_TYPE == tl.float32:
            acc += tl.dot(a, b, allow_tf32=True)
        else:
            acc += tl.dot(a, b)
    
    # Store result to C
    offs_cm = pid_m * BLOCK_M + tl.arange(0, BLOCK_M)
    offs_cn = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)
    
    C_ptr_block = C_batch_ptr + offs_cm[:, None] * stride_Cm + offs_cn[None, :] * stride_Cl
    
    c_mask = (offs_cm[:, None] < M) & (offs_cn[None, :] < L)
    tl.store(C_ptr_block, acc, mask=c_mask)


def triton_batched_matmul(A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
    """Optimized wrapper function with improved grid calculation"""
    assert A.dim() == 3 and B.dim() == 2
    N, M, K = A.shape
    L = B.shape[1]
    
    C = torch.empty((N, M, L), device=A.device, dtype=A.dtype)
    
    # Determine accumulator type and whether to use tensor cores
    if A.dtype == torch.float16:
        acc_dtype = tl.float32
        use_tensor_cores = True
    elif A.dtype == torch.bfloat16:
        acc_dtype = tl.float32
        use_tensor_cores = True
    elif A.dtype == torch.float32:
        acc_dtype = tl.float32
        use_tensor_cores = True
    else:
        acc_dtype = tl.float32
        use_tensor_cores = False
    
    # Calculate grid dimensions - ALWAYS use 3D grid for better occupancy
    grid_m = triton.cdiv(M, 64)  # Use conservative base block size
    grid_n = triton.cdiv(L, 64)
    
    # Ensure we have enough blocks for good occupancy
    total_blocks = N * grid_m * grid_n
    target_min_blocks = 2048  # Target for 128 SMs * 16 blocks/SM
    
    # If grid is too small, reduce block sizes to increase block count
    if total_blocks < target_min_blocks:
        # Increase grid size by using smaller blocks
        grid_m = triton.cdiv(M, 32)
        grid_n = triton.cdiv(L, 32)
        total_blocks = N * grid_m * grid_n
        
        # If still too small, further increase
        if total_blocks < target_min_blocks:
            grid_m = triton.cdiv(M, 16)
            grid_n = triton.cdiv(L, 16)
    
    # Launch kernel with 3D grid
    grid = (N, grid_m, grid_n)
    
    batched_matmul_kernel[grid](
        A, B, C,
        N, M, K, L,
        A.stride(0), A.stride(1), A.stride(2),
        B.stride(0), B.stride(1),
        C.stride(0), C.stride(1), C.stride(2),
        USE_TENSOR_CORES=use_tensor_cores,
        ACC_TYPE=acc_dtype,
    )
    
    return C


class ModelNew(nn.Module):
    def __init__(self):
        super(ModelNew, self).__init__()
    
    def forward(self, A, B):
        return triton_batched_matmul(A, B)
