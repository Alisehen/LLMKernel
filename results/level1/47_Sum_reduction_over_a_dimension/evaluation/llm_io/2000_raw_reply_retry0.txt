import torch, torch.nn as nn, triton, triton.language as tl


@triton.jit
def sum_reduce_lastdim_kernel(
    x_ptr,           # *f32
    y_ptr,           # *f32
    B,               # int32: outer dimension
    N,               # int32: inner dimension
    R,               # int32: reduction dimension (last / contiguous)
    stride_b,        # int32: x.stride(0)
    stride_n,        # int32: x.stride(1)
    stride_r,        # int32: x.stride(2) (should be 1 after layout transform)
    out_stride_b,    # int32: y.stride(0)
    out_stride_n,    # int32: y.stride(1)
    BLOCK_R: tl.constexpr,  # power-of-2 block size along reduction dim
):
    pid = tl.program_id(axis=0)
    b = pid // N
    n = pid % N

    base_in = x_ptr + b * stride_b + n * stride_n
    acc = tl.zeros((), dtype=tl.float32)

    r_offsets = tl.arange(0, BLOCK_R)

    # Loop over reduction dimension in chunks of BLOCK_R
    for r_start in range(0, R, BLOCK_R):
        r_idx = r_start + r_offsets
        mask = r_idx < R

        ptrs = base_in + r_idx * stride_r
        vals = tl.load(ptrs, mask=mask, other=0.0)
        acc += tl.sum(vals, axis=0)

    out_ptr = y_ptr + b * out_stride_b + n * out_stride_n
    tl.store(out_ptr, acc)


def triton_sum_keepdim(x: torch.Tensor, dim: int) -> torch.Tensor:
    """
    Sum reduction with keepdim=True implemented via Triton.

    Optimization: perform a layout transform so that the reduction
    dimension is contiguous (innermost) before launching the Triton kernel.
    """
    assert x.is_cuda, "Input must be a CUDA tensor"
    if x.numel() == 0:
        # Fallback to PyTorch for degenerate case
        return torch.sum(x, dim=dim, keepdim=True)

    orig_shape = x.shape
    ndim = x.ndim

    if dim < 0:
        dim += ndim
    assert 0 <= dim < ndim

    # Make contiguous for predictable strides
    x_contig = x.contiguous()

    # Compute outer, R (reduction size), inner
    sizes = list(orig_shape)
    R = sizes[dim]
    outer = 1
    for i in range(dim):
        outer *= sizes[i]
    inner = 1
    for i in range(dim + 1, ndim):
        inner *= sizes[i]

    # Reshape to (outer, R, inner)
    x_reshaped = x_contig.view(outer, R, inner)

    # Layout transform: move reduction dimension to last and make contiguous
    # Resulting shape: (outer, inner, R) with R contiguous (stride_r = 1)
    x_for_triton = x_reshaped.permute(0, 2, 1).contiguous()
    B = outer
    N = inner

    # Allocate output as (B, N); reduction removes R then we'll reinsert dim
    y_flat = torch.empty((B, N), device=x.device, dtype=x.dtype)

    stride_b, stride_n, stride_r = x_for_triton.stride()
    out_stride_b, out_stride_n = y_flat.stride()

    BLOCK_R = 256  # good choice for R=4096; must be power-of-2

    num_rows = B * N
    grid = lambda meta: (num_rows,)

    sum_reduce_lastdim_kernel[grid](
        x_for_triton,
        y_flat,
        B,
        N,
        R,
        stride_b,
        stride_n,
        stride_r,
        out_stride_b,
        out_stride_n,
        BLOCK_R=BLOCK_R,
        num_warps=4,
    )

    # Reshape back to original shape with keepdim=True
    # y_flat (B, N) -> (outer, 1, inner) -> original out_shape
    y_reshaped = y_flat.view(outer, 1, inner)
    out_shape = list(orig_shape[:dim]) + [1] + list(orig_shape[dim + 1 :])
    y = y_reshaped.view(*out_shape)
    return y


class ModelNew(nn.Module):
    """
    Triton-accelerated model that performs sum reduction over a specified dimension.
    """

    def __init__(self, dim: int):
        super().__init__()
        self.dim = dim

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return triton_sum_keepdim(x, self.dim)
