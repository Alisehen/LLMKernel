import torch, torch.nn as nn, triton, triton.language as tl


@triton.jit
def sum_reduce_dim_kernel(
    x_ptr,       # *f32 / *f16 / *bf16
    out_ptr,     # *f32
    M,           # number of outer batches
    R,           # reduction length
    N,           # number of inner elements
    BLOCK_N: tl.constexpr,
    BLOCK_R: tl.constexpr,
):
    # 3D grid:
    # axis 0 -> tiles along N
    # axis 1 -> batches M
    # axis 2 -> chunks along R (for parallel reduction)
    pid_n = tl.program_id(axis=0)
    pid_m = tl.program_id(axis=1)
    pid_r = tl.program_id(axis=2)

    # Offsets along N
    n_start = pid_n * BLOCK_N
    n_offsets = n_start + tl.arange(0, BLOCK_N)  # [BLOCK_N]
    mask_n = n_offsets < N

    # Single m index
    m = pid_m
    mask_m = m < M

    # Offsets along R for this chunk
    r_start = pid_r * BLOCK_R
    r_offsets = r_start + tl.arange(0, BLOCK_R)  # [BLOCK_R]
    mask_r = r_offsets < R

    # Base offset for this m in [M, R, N] layout
    base_m = m * R * N

    # 2D tile indices: [BLOCK_R, BLOCK_N]
    r_broadcast = r_offsets[:, None]
    n_broadcast = n_offsets[None, :]

    idx = base_m + r_broadcast * N + n_broadcast

    # Validity mask for this tile
    mask_tile = (mask_r[:, None]) & (mask_n[None, :]) & mask_m

    # Load and accumulate along R axis (0) in fp32
    x_tile = tl.load(x_ptr + idx, mask=mask_tile, other=0.0)
    x_tile = x_tile.to(tl.float32)
    tile_sum = tl.sum(x_tile, axis=0)  # [BLOCK_N]

    # Atomic-add partial sums into global output [M, 1, N]
    out_base = m * N + n_offsets
    mask_out = mask_n & mask_m
    tl.atomic_add(out_ptr + out_base, tile_sum, mask=mask_out)


def triton_sum_keepdim(x: torch.Tensor, dim: int) -> torch.Tensor:
    """Sum reduction over a given dim with keepdim=True using Triton."""
    # Fallback for non-CUDA or unsupported dtypes
    if (not x.is_cuda) or (x.dtype not in (torch.float16, torch.bfloat16, torch.float32)):
        return torch.sum(x, dim=dim, keepdim=True)

    ndim = x.ndim
    if dim < 0:
        dim += ndim
    assert 0 <= dim < ndim

    orig_shape = list(x.shape)

    # Flatten to [M, R, N] where we reduce over R (axis=1)
    if dim == 0:
        M = 1
        R = orig_shape[0]
        if ndim > 1:
            N = int(torch.prod(torch.tensor(orig_shape[1:], device="cpu")))
        else:
            N = 1
    elif dim == ndim - 1:
        if ndim > 1:
            M = int(torch.prod(torch.tensor(orig_shape[:-1], device="cpu")))
        else:
            M = 1
        R = orig_shape[-1]
        N = 1
    else:
        M = int(torch.prod(torch.tensor(orig_shape[:dim], device="cpu")))
        R = orig_shape[dim]
        if dim + 1 < ndim:
            N = int(torch.prod(torch.tensor(orig_shape[dim + 1 :], device="cpu")))
        else:
            N = 1

    # Ensure contiguous before reshape for predictable layout
    x_contig = x.contiguous()
    x_reshaped = x_contig.reshape(M, R, N)

    # Accumulate in fp32 for numerical stability
    out_reshaped = torch.empty((M, 1, N), device=x.device, dtype=torch.float32)
    # We'll use atomic_add in the kernel, so initialize to zero
    out_reshaped.zero_()

    # Tunable block sizes (power-of-two as required)
    BLOCK_N = 128
    BLOCK_R = 128

    def grid(meta):
        # 3D grid:
        #   axis 0: tiles along N
        #   axis 1: M
        #   axis 2: chunks along R (for parallel reduction)
        return (
            triton.cdiv(N, meta["BLOCK_N"]),  # along N
            max(1, M),                        # along M
            triton.cdiv(R, meta["BLOCK_R"]),  # along R (chunks)
        )

    sum_reduce_dim_kernel[grid](
        x_reshaped,
        out_reshaped,
        M,
        R,
        N,
        BLOCK_N=BLOCK_N,
        BLOCK_R=BLOCK_R,
        num_warps=4,
        num_stages=2,
    )

    # Cast back to original dtype if needed
    if x.dtype != torch.float32:
        out_reshaped = out_reshaped.to(x.dtype)

    # Reshape back to original shape with reduced dim kept as size 1
    out_shape = orig_shape.copy()
    out_shape[dim] = 1
    return out_reshaped.reshape(out_shape)
