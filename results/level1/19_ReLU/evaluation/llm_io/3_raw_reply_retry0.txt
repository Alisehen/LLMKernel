import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def relu_kernel_optimized(
    x_ptr,
    output_ptr,
    n_elements,
    BLOCK_SIZE: tl.constexpr,
    VECTOR_SIZE: tl.constexpr,
    NUM_STAGES: tl.constexpr,
):
    """Optimized ReLU kernel with vectorized loads/stores and latency hiding."""
    pid = tl.program_id(axis=0)
    num_vectors = BLOCK_SIZE // VECTOR_SIZE
    
    # Vectorized offsets
    vector_offset = pid * num_vectors
    vector_indices = vector_offset + tl.arange(0, num_vectors)
    vector_offsets = vector_indices * VECTOR_SIZE
    
    # Create mask for the entire vector
    vector_mask = vector_offsets[:, None] + tl.arange(0, VECTOR_SIZE)[None, :] < n_elements
    flat_mask = vector_mask.flatten()
    flat_offsets = (vector_offsets[:, None] + tl.arange(0, VECTOR_SIZE)[None, :]).flatten()
    
    # Vectorized load
    x_vector = tl.load(x_ptr + flat_offsets, mask=flat_mask, cache_modifier=".cg")
    
    # Fast ReLU: (x + |x|) / 2
    abs_x = tl.abs(x_vector)
    output = (x_vector + abs_x) * 0.5
    
    # Vectorized store
    tl.store(output_ptr + flat_offsets, output, mask=flat_mask)

            num_warps=config['num_warps']
        )
        break  # Use first valid config for now

class ModelNew(nn.Module):
