import torch

import triton.language as tl


@triton.jit
def relu_kernel(
    x_ptr,
    output_ptr,
    n_elements,
    BLOCK_SIZE: tl.constexpr,
):
    """Triton kernel for ReLU activation."""
    pid = tl.program_id(axis=0)
    block_start = pid * BLOCK_SIZE
    offsets = block_start + tl.arange(0, BLOCK_SIZE)
    mask = offsets < n_elements
    
    x = tl.load(x_ptr + offsets, mask=mask)
    # ReLU implementation: max(x, 0)
    # Use bitwise operations for efficiency: zero = 0.0
    # Compute mask for positive values using comparison
    positive_mask = x > 0.0
    output = tl.where(positive_mask, x, 0.0)
    
    tl.store(output_ptr + offsets, output, mask=mask)

    abs_x = tl.abs(x)
    output = (x + abs_x) * 0.5
    
    tl.store(output_ptr + offsets, output, mask=mask)


def triton_relu(x: torch.Tensor) -> torch.Tensor:

    output = torch.empty_like(x)
