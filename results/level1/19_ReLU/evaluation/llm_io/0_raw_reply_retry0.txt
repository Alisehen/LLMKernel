import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def relu_kernel(
    x_ptr,
    output_ptr,
    n_elements,
    BLOCK_SIZE: tl.constexpr,
):
    """Triton kernel for ReLU activation."""
    pid = tl.program_id(axis=0)
    block_start = pid * BLOCK_SIZE
    offsets = block_start + tl.arange(0, BLOCK_SIZE)
    mask = offsets < n_elements
    
    x = tl.load(x_ptr + offsets, mask=mask)
    # ReLU implementation: max(x, 0)
    # Use bitwise operations for efficiency: zero = 0.0
    # Compute mask for positive values using comparison
    positive_mask = x > 0.0
    output = tl.where(positive_mask, x, 0.0)
    
    tl.store(output_ptr + offsets, output, mask=mask)


@triton.jit
def relu_kernel_fast(
    x_ptr,
    output_ptr,
    n_elements,
    BLOCK_SIZE: tl.constexpr,
):
    """Optimized ReLU kernel using arithmetic operations."""
    pid = tl.program_id(axis=0)
    block_start = pid * BLOCK_SIZE
    offsets = block_start + tl.arange(0, BLOCK_SIZE)
    mask = offsets < n_elements
    
    x = tl.load(x_ptr + offsets, mask=mask)
    # Fast ReLU using absolute value: (x + |x|) / 2
    # Avoids branching and uses efficient arithmetic
    abs_x = tl.abs(x)
    output = (x + abs_x) * 0.5
    
    tl.store(output_ptr + offsets, output, mask=mask)


def triton_relu(x: torch.Tensor) -> torch.Tensor:
    """Triton wrapper for ReLU activation."""
    output = torch.empty_like(x)
    n_elements = output.numel()
    
    # Heuristic for block size selection
    # Use larger blocks for better memory coalescing on modern GPUs
    if n_elements >= 2**20:  # 1M+ elements
        BLOCK_SIZE = 1024
    elif n_elements >= 2**14:  # 16K+ elements  
        BLOCK_SIZE = 512
    else:
        BLOCK_SIZE = 256
    
    # Ensure BLOCK_SIZE is power of 2
    BLOCK_SIZE = 1 << (BLOCK_SIZE.bit_length() - 1)
    
    grid = (triton.cdiv(n_elements, BLOCK_SIZE),)
    
    # Choose kernel based on tensor size and dtype
    if x.dtype == torch.float32 and n_elements >= 2**16:
        # Use fast kernel for large FP32 tensors
        relu_kernel_fast[grid](x, output, n_elements, BLOCK_SIZE=BLOCK_SIZE)
    else:
        # Standard kernel for other cases
        relu_kernel[grid](x, output, n_elements, BLOCK_SIZE=BLOCK_SIZE)
    
    return output


class ModelNew(nn.Module):
    """Optimized model using Triton kernels for ReLU activation."""
    def __init__(self):
        super(ModelNew, self).__init__()
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return triton_relu(x)
