import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.autotune(
    configs=[
        triton.Config({"BLOCK": 1}, num_warps=1, num_stages=2),
        triton.Config({"BLOCK": 2}, num_warps=2, num_stages=3),
        triton.Config({"BLOCK": 4}, num_warps=4, num_stages=3),
        triton.Config({"BLOCK": 8}, num_warps=4, num_stages=2),
        triton.Config({"BLOCK": 16}, num_warps=8, num_stages=3),
    ],
    key=["axis_len"],
)
@triton.jit
def cumprod_kernel(
    x_ptr,
    y_ptr,
    axis_len,
    inner,
    seq_count,
    BLOCK: tl.constexpr,
):
    pid = tl.program_id(axis=0)
    if pid >= seq_count:
        return

    axis_len_val = axis_len
    if axis_len_val <= 0:
        return

    inner_val = inner
    seq_stride = axis_len_val * inner_val
    outer_idx = pid // inner_val
    inner_idx = pid % inner_val
    base_offset = outer_idx * seq_stride + inner_idx

    first_offset = base_offset
    first_val = tl.load(x_ptr + first_offset)
    tl.store(y_ptr + first_offset, first_val)
    running = first_val

    start = 1
    next_offset = first_offset + inner_val

    while start < axis_len_val:
        cur_offset = next_offset
        for i in tl.static_range(0, BLOCK):
            idx = start + i
            mask = idx < axis_len_val
            ptr_x = x_ptr + cur_offset
            ptr_y = y_ptr + cur_offset
            val = tl.load(ptr_x, mask=mask, other=1)
            running = tl.where(mask, running * val, running)
            tl.store(ptr_y, running, mask=mask)
            cur_offset += inner_val
        next_offset = cur_offset
        start += BLOCK


def triton_cumprod(x: torch.Tensor, dim: int) -> torch.Tensor:
    if not x.is_cuda:
        raise RuntimeError("Input must be a CUDA tensor.")
    x_contig = x.contiguous()
    out = torch.empty_like(x_contig)

    ndim = x_contig.ndim
    dim = dim if dim >= 0 else dim + ndim
    axis_len = x_contig.shape[dim]
    if axis_len == 0:
        return out

    inner = x_contig.stride(dim)
    seq_count = x_contig.numel() // axis_len
    grid = (seq_count,)

    cumprod_kernel[grid](
        x_contig,
        out,
        axis_len,
        inner,
        seq_count,
    )
    return out


class ModelNew(nn.Module):
    def __init__(self, dim):
        super().__init__()
        self.dim = dim

    def forward(self, x):
        return triton_cumprod(x, self.dim)
