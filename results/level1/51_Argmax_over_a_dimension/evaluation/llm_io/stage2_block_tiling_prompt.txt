# torch, torch.nn, triton, triton.language as tl
import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def argmax_kernel(
    x_ptr,
    out_ptr,
    outer_size,
    reduce_size,
    inner_size,
    BLOCK_INNER: tl.constexpr,
    BLOCK_K: tl.constexpr,
):
    pid_inner = tl.program_id(0)
    pid_outer = tl.program_id(1)

    if pid_outer >= outer_size:
        return

    inner_tile_start = pid_inner * BLOCK_INNER
    inner_offsets = inner_tile_start + tl.arange(0, BLOCK_INNER)
    inner_mask = inner_offsets < inner_size

    inner_offsets_i64 = inner_offsets.to(tl.int64)
    inner_size_i64 = tl.full((), inner_size, tl.int64)
    reduce_size_i64 = tl.full((), reduce_size, tl.int64)
    outer_idx_i64 = pid_outer.to(tl.int64)

    slice_base = outer_idx_i64 * reduce_size_i64 * inner_size_i64
    out_base = outer_idx_i64 * inner_size_i64 + inner_offsets_i64

    best_val = tl.full((BLOCK_INNER,), -float("inf"), tl.float32)
    best_idx = tl.full((BLOCK_INNER,), 0, tl.int32)

    k_range = tl.arange(0, BLOCK_K)

    k = 0
    while k < reduce_size:
        offsets = k + k_range
        offsets_i64 = offsets.to(tl.int64)
        k_mask = offsets < reduce_size

        ptrs = slice_base + offsets_i64[:, None] * inner_size_i64 + inner_offsets_i64[None, :]
        mask = k_mask[:, None] & inner_mask[None, :]

        vals = tl.load(x_ptr + ptrs, mask=mask, other=-float("inf")).to(tl.float32)

        current_max = tl.max(vals, axis=0)
        current_arg = tl.argmax(vals, axis=0)
        current_idx = current_arg + k

        better = current_max > best_val
        best_val = tl.where(better, current_max, best_val)
        best_idx = tl.where(better, current_idx, best_idx)

        k += BLOCK_K

    tl.store(out_ptr + out_base, best_idx.to(tl.int64), mask=inner_mask)


def triton_argmax(x: torch.Tensor, dim: int) -> torch.Tensor:
    x = x.contiguous()
    ndim = x.ndim
    if dim < 0:
        dim += ndim

    reduce_size = x.shape[dim]

    outer = 1
    for i in range(dim):
        outer *= x.shape[i]

    inner = 1
    for i in range(dim + 1, ndim):
        inner *= x.shape[i]

    out_shape = x.shape[:dim] + x.shape[dim + 1 :]
    out = torch.empty(out_shape, dtype=torch.int64, device=x.device)
    n_slices = out.numel()
    if n_slices == 0:
        return out

    BLOCK_INNER = 32
    BLOCK_K = 256

    grid = lambda META: (
        triton.cdiv(inner, META["BLOCK_INNER"]),
        outer,
    )

    argmax_kernel[grid](
        x,
        out.view(-1),
        outer,
        reduce_size,
        inner,
        BLOCK_INNER=BLOCK_INNER,
        BLOCK_K=BLOCK_K,
        num_warps=4,
        num_stages=2,
    )
    return out


class ModelNew(nn.Module):
