@triton.jit
def max_partial_kernel(
    x_ptr,            # *f32 or *f16

    M, N,             # rows, reduction length

    stride_pm, stride_pn,
    BLOCK_N: tl.constexpr,

    First-stage reduction: compute per-block maxima along the last dimension.
    Grid:
