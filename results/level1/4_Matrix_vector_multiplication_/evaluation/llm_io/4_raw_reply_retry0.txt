```python
import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.autotune(
    configs=[
        triton.Config({'BLOCK_M': 128, 'BLOCK_K': 256}, num_warps=4, num_stages=4),
        triton.Config({'BLOCK_M': 256, 'BLOCK_K': 128}, num_warps=8, num_stages=3),
        triton.Config({'BLOCK_M': 64, 'BLOCK_K': 256}, num_warps=4, num_stages=4),
        triton.Config({'BLOCK_M': 128, 'BLOCK_K': 128}, num_warps=4, num_stages=3),
    ],
    key=['M', 'K'],
)
@triton.jit
def matvec_kernel(
    A_ptr,
    B_ptr,
    C_ptr,
    M,
    K,
    stride_am,
    stride_ak,
    stride_bk,
    stride_cm,
    BLOCK_M: tl.constexpr,
    BLOCK_K: tl.constexpr,
):
    # 1D grid over rows - each program processes BLOCK_M rows
    pid = tl.program_id(0)
    
    # Offsets for this block
    offs_m = pid * BLOCK_M + tl.arange(0, BLOCK_M)
    m_mask = offs_m < M
    
    # Initialize accumulator
    acc = tl.zeros((BLOCK_M,), dtype=tl.float32)
    
    # Pre-fetch first B block to hide latency
    k_start = 0
    offs_k_start = tl.arange(0, BLOCK_K)
    k_mask_start = offs_k_start < K
    B_ptrs_start = B_ptr + offs_k_start * stride_bk
    b_start = tl.load(B_ptrs_start, mask=k_mask_start)
    
    # Loop over K dimension in blocks
    for k in range(0, tl.cdiv(K, BLOCK_K)):
        # Compute current K offset
        offs_k = k * BLOCK_K + tl.arange(0, BLOCK_K)
        k_mask = offs_k < K
        
        # Load next B block (pre-fetch for next iteration)
        next_k = k + 1
        if next_k * BLOCK_K < K:
            next_offs_k = next_k * BLOCK_K + tl.arange(0, BLOCK_K)
            next_k_mask = next_offs_k < K
            next_B_ptrs = B_ptr + next_offs_k * stride_bk
            b_next = tl.load(next_B_ptrs, mask=next_k_mask)
        else:
            b_next = tl.zeros((BLOCK_K,), dtype=tl.float32)
        
        # Load A block: (BLOCK_M, BLOCK_K)
        A_ptrs = A_ptr + offs_m[:, None] * stride_am + offs_k[None, :] * stride_ak
        a = tl.load(A_ptrs, mask=m_mask[:, None] & k_mask[None, :])
        
        # Use pre-fetched B
        b = tl.where(k == 0, b_start, b_prev)
        
        # Accumulate dot product with manual unrolling hint
        # Using manual reduction for better control
        acc += tl.sum(a * b[None, :], axis=1)
        
        # Prepare for next iteration
        b_prev = b_next
    
    # Store result
    C_ptrs = C_ptr + offs_m * stride_cm
    tl.store(C_ptrs, acc, mask=m_mask)


def triton_matvec(A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
    M, K = A.shape
    assert B.shape == (K, 1), f"B must be shape ({K}, 1), got {B.shape}"
    
    # Allocate output tensor
    C = torch.empty((M, 1), device=A.device, dtype=A.dtype)
    
    # Convert to float32 for accumulation
    A_acc = A if A.dtype == torch.float32 else A.to(torch.float32)
    B_acc = B if B.dtype == torch.float32 else B.to(torch.float32)
    C_acc = C if C.dtype == torch.float32 else C.to(torch.float32)
    
    # Grid configuration - 1D over rows
    grid = lambda META: (triton.cdiv(M, META['BLOCK_M']),)
    
    # Launch kernel
    matvec_kernel[grid](
        A_acc,
        B_acc,
        C_acc,
        M,
        K,
        A_acc.stride(0),
        A_acc.stride(1),
        B_acc.stride(0),
        C_acc.stride(0),
    )
    
    # Convert back if needed
    if C.dtype != torch.float32:
        C.copy_(C_acc.to(C.dtype))
    
    return C


class ModelNew(nn.Module):
    """
    Simple model that performs matrix-vector multiplication (C = A * B).
    """
    def __init__(self):
        super(ModelNew, self).__init__()
    
    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        """
        Performs matrix-vector multiplication using optimized Triton kernels.

        Args:
            A: Input matrix of shape (M, K).
            B: Input vector of shape (K, 1).

        Returns:
            Output vector of shape (M, 1).
        """
        return triton_matvec(A, B)
```

## Optimization Summary

Based on NCU metrics showing low warp occupancy (8.33%) and high DRAM throughput (84.83%), I made these key optimizations:

1. **Balanced Tile Sizes**: Adjusted autotune configs to focus on BLOCK_M=128/256 and BLOCK_K=128/256 combinations. These maintain power-of-2 constraints while improving workload balance.

2. **Prefetching**: Added B vector prefetching to hide memory latency. The kernel now loads the next B block while computing with the current one.

3. **Memory Access Optimization**: Maintained coalesced memory access patterns while adding explicit control flow for boundary conditions.

4. **Resource Utilization**: 
   - Increased `num_stages` to 4 for smaller blocks to improve instruction-level parallelism
   - Used `num_warps=8` for larger BLOCK_M=256 to better utilize SM resources

5. **Precision Handling**: Explicit float32 accumulation with proper dtype conversion to ensure numerical stability while allowing input flexibility.

The configs prioritize:
- BLOCK_M=128, BLOCK_K=256 (balanced load, good for tall matrices)
- BLOCK_M=256, BLOCK_K=128 (better for wider matrices, higher thread count)
- Fallback options for different M/K ratios

These changes should significantly improve warp occupancy and SM throughput while maintaining the memory-bound nature of the matvec operation.
