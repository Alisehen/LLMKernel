def matmul_triu_kernel(

def triton_matmul_triu(A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:

    B: (K, N)
    Returns: (M, N) upper-triangular result.

    assert A.is_cuda, "Triton matmul requires CUDA tensors"

    B_mat = B.contiguous()

        BLOCK_K=BLOCK_K,
        num_warps=8,
