import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def softsign_kernel_optimized(
    x_ptr,
    output_ptr,
    n_elements,
    BLOCK_SIZE: tl.constexpr,
    VECTOR_SIZE: tl.constexpr,
    USE_FAST_DIV: tl.constexpr,
    num_stages: tl.constexpr,
):
    """
    Optimized Softsign kernel with vectorized memory access and configurable pipeline stages.
    
    Key optimizations:
    1. Vectorized loads/stores for better memory coalescing
    2. Configurable pipeline stages to hide memory latency
    3. Optimized division based on data type
    4. Reduced register pressure through smart operation ordering
    """
    pid = tl.program_id(axis=0)
    block_start = pid * (BLOCK_SIZE * VECTOR_SIZE)
    offsets = block_start + tl.arange(0, BLOCK_SIZE)[:, None] * VECTOR_SIZE + tl.arange(0, VECTOR_SIZE)[None, :]
    offsets_flat = tl.view(offsets, BLOCK_SIZE * VECTOR_SIZE)
    
    # Create mask for the entire vectorized block
    mask_flat = offsets_flat < n_elements
    mask_reshaped = tl.view(mask_flat, (BLOCK_SIZE, VECTOR_SIZE))
    
    # Vectorized load with proper mask
    x_vec = tl.load(x_ptr + offsets, mask=mask_reshaped, other=0.0)
    
    # Compute absolute value (fused operations)
    abs_x_vec = tl.abs(x_vec)
    
    # Optimized computation based on USE_FAST_DIV flag
    if USE_FAST_DIV:
        # Use native division for Tensor Core architectures
        # Compiler optimizes this pattern for Ampere/Ada
        output_vec = x_vec * tl.math.rcp(1.0 + abs_x_vec)
    else:
        # Standard computation with fused multiply-add
        denominator = 1.0 + abs_x_vec
        output_vec = x_vec / denominator
    
    # Vectorized store
    tl.store(output_ptr + offsets, output_vec, mask=mask_reshaped)

@triton.jit
def softsign_kernel_autotuned(
    x_ptr,
    output_ptr,
    n_elements,
    BLOCK_SIZE: tl.constexpr,
    VECTOR_SIZE: tl.constexpr,
    USE_FAST_DIV: tl.constexpr,
    num_stages: tl.constexpr,
):
    """Autotuned version with optimized configurations for different tensor sizes."""
    pid = tl.program_id(axis=0)
    block_elems = BLOCK_SIZE * VECTOR_SIZE
    block_start = pid * block_elems
    
    # Vectorized offsets calculation
    offsets = block_start + tl.arange(0, BLOCK_SIZE)[:, None] * VECTOR_SIZE + tl.arange(0, VECTOR_SIZE)[None, :]
    offsets_flat = tl.reshape(offsets, block_elems)
    
    # Vectorized mask
    mask_flat = offsets_flat < n_elements
    mask = tl.reshape(mask_flat, (BLOCK_SIZE, VECTOR_SIZE))
    
    # Coalesced vector load
    x_vec = tl.load(x_ptr + offsets, mask=mask, other=0.0)
    
    # Compute with minimal register pressure
    if USE_FAST_DIV:
        # Optimized for Tensor Cores: x * rcp(1 + |x|)
        abs_x = tl.abs(x_vec)
        output_vec = x_vec * tl.math.rcp(1.0 + abs_x)
    else:
        # Standard precision: x / (1 + |x|)
        output_vec = x_vec / (1.0 + tl.abs(x_vec))
    
    # Coalesced vector store
    tl.store(output_ptr + offsets, output_vec, mask=mask)


def triton_softsign(x: torch.Tensor) -> torch.Tensor:
    """
    Optimized Triton wrapper for Softsign activation.
    
    Uses autotuned kernel configurations for optimal performance across:
    - Tensor sizes (small to large)
    - Data types (FP32, FP16, BF16)
    - Memory access patterns
    """
    output = torch.empty_like(x)
    n_elements = output.numel()
    
    if n_elements == 0:
        return output
    
    # Calculate optimal grid size
    grid = lambda meta: (triton.cdiv(n_elements, meta['BLOCK_SIZE'] * meta['VECTOR_SIZE']),)
    
    # Determine if we should use fast division (optimized for Tensor Cores)
    use_fast_div = x.dtype in (torch.float16, torch.bfloat16)
    
    # Use autotuned kernel for best performance
    softsign_kernel_autotuned[grid](
        x, output, n_elements,
        USE_FAST_DIV=use_fast_div,
        dtype=x.dtype
    )
    
    return output
