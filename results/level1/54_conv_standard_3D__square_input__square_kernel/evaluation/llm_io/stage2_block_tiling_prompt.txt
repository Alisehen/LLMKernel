import triton
import triton.language as tl

@triton.jit
def conv3d_fwd_gemm_kernel(
    x_ptr,  # float32[N, Cin, D, H, W]
    w_ptr,  # float32[Cout, K_dim] with K_dim = Cin_per_group * KD * KH * KW
    b_ptr,  # float32[Cout] or dummy if HAS_BIAS=False
    y_ptr,  # float32[N, Cout, D_out, H_out, W_out]
    #
    N,
    Cin,
    D,
    H,
    W,
    Cout,
    D_out,
    H_out,
    W_out,
    Cin_per_group,
    Cout_per_group,
    groups,
    stride_d,
    stride_h,
    stride_w,
    pad_d,
    pad_h,
    pad_w,
    dil_d,
    dil_h,
    dil_w,
    x_stride_n,
    x_stride_c,
    x_stride_d,
    x_stride_h,
    x_stride_w,
    y_stride_n,
    y_stride_c,
    y_stride_d,
    y_stride_h,
    y_stride_w,
    #
    KD: tl.constexpr,
    KH: tl.constexpr,
    KW: tl.constexpr,
    K_dim: tl.constexpr,
    BLOCK_M: tl.constexpr,
    BLOCK_N: tl.constexpr,
    BLOCK_K: tl.constexpr,
    HAS_BIAS: tl.constexpr,
):
    # 3D grid:
    #  pid_m  : tiles over output positions (flattened N*D_out*H_out*W_out)
    #  pid_t  : tiles over output channels within a group
    #  pid_g  : group index
    pid_m = tl.program_id(axis=0)
    pid_t = tl.program_id(axis=1)
    pid_g = tl.program_id(axis=2)

    # Total number of output positions (flattened)
    M_dim = N * D_out * H_out * W_out

    # -------------------------
    # Decode M tile (output positions)
    # -------------------------
    m_start = pid_m * BLOCK_M
    m_offsets = m_start + tl.arange(0, BLOCK_M)
    m_mask = m_offsets < M_dim

    # Decode flattened index m_offsets -> (n_idx, d_out_idx, h_out_idx, w_out_idx)
    tmp = m_offsets
    w_out_idx = tmp % W_out
    tmp = tmp // W_out
    h_out_idx = tmp % H_out
    tmp = tmp // H_out
    d_out_idx = tmp % D_out
    n_idx = tmp // D_out

    # Base input coordinates for each output position in the tile
    in_z_base = d_out_idx * stride_d - pad_d
    in_y_base = h_out_idx * stride_h - pad_h
    in_x_base = w_out_idx * stride_w - pad_w

    # -------------------------
    # Decode N tile (output channels) with grouping
    # -------------------------
    oc_local_start = pid_t * BLOCK_N
    oc_local_offsets = oc_local_start + tl.arange(0, BLOCK_N)
    oc_local_mask = oc_local_offsets < Cout_per_group

    # Global output channel indices
    oc_offsets = pid_g * Cout_per_group + oc_local_offsets
    oc_mask = oc_local_mask  # groups * Cout_per_group == Cout

    # -------------------------
    # Initialize accumulator
    # -------------------------
    acc = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)

    # -------------------------
    # K-dimension decomposition
    # K_dim = Cin_per_group * KD * KH * KW
    # -------------------------
    KDHW = KD * KH * KW
    KHW = KH * KW

    # Loop over K in BLOCK_K chunks (implicit GEMM)
    for k_start in range(0, K_dim, BLOCK_K):
        k_idx = k_start + tl.arange(0, BLOCK_K)  # [BLOCK_K]
        k_mask = k_idx < K_dim

        # Decode k_idx -> (cin_inner, kd_idx, kh_idx, kw_idx)
        cin_inner = k_idx // KDHW
        rem = k_idx % KDHW
        kd_idx = rem // KHW
        rem2 = rem % KHW
        kh_idx = rem2 // KW
        kw_idx = rem2 % KW

        # Input channel indices for this group: [BLOCK_K]
        in_c = pid_g * Cin_per_group + cin_inner

        # Kernel spatial offsets scaled by dilation: [BLOCK_K]
        kz = kd_idx * dil_d
        ky = kh_idx * dil_h
        kx = kw_idx * dil_w

        # Input spatial coordinates: [BLOCK_M, BLOCK_K]
        in_z = in_z_base[:, None] + kz[None, :]
        in_y = in_y_base[:, None] + ky[None, :]
        in_x = in_x_base[:, None] + kx[None, :]

        # Bounds check for input
        in_bounds = (
            (in_z >= 0)
            & (in_z < D)
            & (in_y >= 0)
            & (in_y < H)
            & (in_x >= 0)
            & (in_c[None, :] >= 0)
            & (in_c[None, :] < Cin)
        )

        # Compute input pointer offsets: [BLOCK_M, BLOCK_K]
        x_offsets = (
            n_idx[:, None] * x_stride_n
            + in_c[None, :] * x_stride_c
            + in_z * x_stride_d
            + in_y * x_stride_h
            + in_x * x_stride_w
        )

        load_mask_x = m_mask[:, None] & k_mask[None, :] & in_bounds
        x_vals = tl.load(x_ptr + x_offsets, mask=load_mask_x, other=0.0)

        # Load weight tile: [BLOCK_K, BLOCK_N]
        k_broadcast = k_idx[:, None]
        oc_broadcast = oc_offsets[None, :]
        w_offsets = oc_broadcast * K_dim + k_broadcast
        w_mask = k_mask[:, None] & oc_mask[None, :]
        w_vals = tl.load(w_ptr + w_offsets, mask=w_mask, other=0.0)

        # GEMM: [BLOCK_M, BLOCK_K] x [BLOCK_K, BLOCK_N] -> [BLOCK_M, BLOCK_N]
        acc += tl.dot(x_vals, w_vals, allow_tf32=True)

    # -------------------------
    # Add bias (broadcast over M)
    # -------------------------
    if HAS_BIAS:
        bias_vals = tl.load(b_ptr + oc_offsets, mask=oc_mask, other=0.0)  # [BLOCK_N]
        acc += bias_vals[None, :]

    # -------------------------
    # Store results to y
    # -------------------------
    y_offsets = (
        n_idx[:, None] * y_stride_n
        + oc_offsets[None, :] * y_stride_c
        + d_out_idx[:, None] * y_stride_d
        + h_out_idx[:, None] * y_stride_h
        + w_out_idx[:, None] * y_stride_w
    )

    store_mask = m_mask[:, None] & oc_mask[None, :]
    tl.store(y_ptr + y_offsets, acc, mask=store_mask)


def triton_conv3d(
    x: torch.Tensor,
    weight: torch.Tensor,
    bias: torch.Tensor,
    stride,
    padding,
    dilation,
    groups: int,
) -> torch.Tensor:
    """
    High-performance 3D convolution using Triton, via implicit GEMM.

    Args:
        x: Input tensor of shape [N, Cin, D, H, W], float32, CUDA.
        weight: [Cout, Cin/groups, KD, KH, KW], float32, CUDA.
        bias: [Cout] or None.
        stride, padding, dilation: int or 3-tuple.
        groups: number of groups.

    Returns:
        y: [N, Cout, D_out, H_out, W_out], float32, CUDA.
    """
    assert x.is_cuda and weight.is_cuda, "Inputs must be on CUDA"
    assert x.dtype == torch.float32 and weight.dtype == torch.float32, "Only float32 supported"
    assert x.ndim == 5 and weight.ndim == 5, "x must be [N,C,D,H,W], weight [Cout,Cin/groups,KD,KH,KW]"

    N, Cin, D, H, W = x.shape
    Cout, Cin_per_group, KD, KH, KW = weight.shape
    assert Cin == Cin_per_group * groups, "Cin must equal Cin_per_group * groups"
    assert Cout > 0, "Cout must be positive"
    if bias is not None:
        assert bias.shape[0] == Cout

    # Normalize stride/padding/dilation to 3-tuples
    if isinstance(stride, int):
        stride_d = stride_h = stride_w = stride
    else:
        stride_d, stride_h, stride_w = stride
    if isinstance(padding, int):
        pad_d = pad_h = pad_w = padding
    else:
        pad_d, pad_h, pad_w = padding
    if isinstance(dilation, int):
        dil_d = dil_h = dil_w = dilation
    else:
        dil_d, dil_h, dil_w = dilation

    # Output dimensions (match PyTorch Conv3d)
    D_out = (D + 2 * pad_d - dil_d * (KD - 1) - 1) // stride_d + 1
    H_out = (H + 2 * pad_h - dil_h * (KH - 1) - 1) // stride_h + 1
    W_out = (W + 2 * pad_w - dil_w * (KW - 1) - 1) // stride_w + 1
    assert D_out > 0 and H_out > 0 and W_out > 0, "Output spatial dims must be positive"

    # Make contiguous tensors
    x_contig = x.contiguous()
    w_contig = weight.contiguous()
    # Flatten weight to [Cout, K_dim]
    K_dim = Cin_per_group * KD * KH * KW
    w_flat = w_contig.view(Cout, K_dim)
    if bias is not None:
        b_contig = bias.contiguous()
    else:
        # Dummy tensor; won't be used when HAS_BIAS=False
        b_contig = torch.empty(1, device=x.device, dtype=x.dtype)

    y = torch.empty((N, Cout, D_out, H_out, W_out), device=x.device, dtype=x.dtype)

    # Strides
    x_s = x_contig.stride()
    y_s = y.stride()
    x_stride_n, x_stride_c, x_stride_d, x_stride_h, x_stride_w = x_s
    y_stride_n, y_stride_c, y_stride_d, y_stride_h, y_stride_w = y_s

    Cout_per_group = Cout // groups
    M_dim = N * D_out * H_out * W_out

    # 3D grid:
    #   axis 0: tiles over flattened output positions M_dim
    #   axis 1: tiles over output channels within each group
    #   axis 2: groups
    def grid(meta):
        return (
            triton.cdiv(M_dim, meta["BLOCK_M"]),
            triton.cdiv(Cout_per_group, meta["BLOCK_N"]),
            groups,
        )

    conv3d_fwd_gemm_kernel[grid](
        x_contig,
        w_flat,
        b_contig,
        y,
        #
        N,
        Cin,
        D,
        H,
        W,
        Cout,
        D_out,
        H_out,
        W_out,
        Cin_per_group,
        Cout_per_group,
        groups,
        stride_d,
        stride_h,
        stride_w,
        pad_d,
        pad_h,
        pad_w,
        dil_d,
        dil_h,
        dil_w,
        x_stride_n,
        x_stride_c,
        x_stride_d,
        x_stride_h,
        x_stride_w,
        y_stride_n,
        y_stride_c,
        y_stride_d,
        y_stride_h,
        y_stride_w,
        #
        KD=KD,
        KH=KH,
        KW=KW,
        K_dim=K_dim,
        HAS_BIAS=bias is not None,
    )

    return y
