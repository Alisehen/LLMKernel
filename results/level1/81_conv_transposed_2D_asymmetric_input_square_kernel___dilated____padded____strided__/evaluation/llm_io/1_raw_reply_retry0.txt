import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def conv_transpose2d_kernel(
    x_ptr,
    w_ptr,
    bias_ptr,
    y_ptr,
    N,
    IC,
    IH,
    IW,
    OC,
    KH,
    KW,
    OH,
    OW,
    STRIDE_H,
    STRIDE_W,
    PAD_H,
    PAD_W,
    DIL_H,
    DIL_W,
    K,
    ADD_BIAS: tl.constexpr,
    BLOCK_M: tl.constexpr,
    BLOCK_N: tl.constexpr,
    BLOCK_K: tl.constexpr,
):
    pid_n = tl.program_id(axis=0)
    pid_m = tl.program_id(axis=1)
    pid_o = tl.program_id(axis=2)

    ohow = OH * OW
    m_offsets = pid_m * BLOCK_M + tl.arange(0, BLOCK_M)
    n_offsets = pid_o * BLOCK_N + tl.arange(0, BLOCK_N)

    m_mask = m_offsets < ohow
    n_mask = n_offsets < OC

    m_offsets_int = m_offsets.to(tl.int32)
    oh_idx = m_offsets_int // OW
    ow_idx = m_offsets_int - oh_idx * OW

    acc = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)

    for k_start in range(0, K, BLOCK_K):
        k_offsets = k_start + tl.arange(0, BLOCK_K)
        k_mask = k_offsets < K

        k_offsets_int = k_offsets.to(tl.int32)
        ic_span = KH * KW
        ic_idx = k_offsets_int // ic_span
        kt = k_offsets_int - ic_idx * ic_span
        kh_idx = kt // KW
        kw_idx = kt - kh_idx * KW

        oh_pad = oh_idx[:, None] + PAD_H
        ow_pad = ow_idx[:, None] + PAD_W

        ih_nom = oh_pad - kh_idx[None, :] * DIL_H
        iw_nom = ow_pad - kw_idx[None, :] * DIL_W

        ih_div = ih_nom // STRIDE_H
        iw_div = iw_nom // STRIDE_W

        valid_h = (ih_nom >= 0) & (ih_div * STRIDE_H == ih_nom) & (ih_div < IH)
        valid_w = (iw_nom >= 0) & (iw_div * STRIDE_W == iw_nom) & (iw_div < IW)
        valid = m_mask[:, None] & k_mask[None, :] & valid_h & valid_w

        safe_ih = tl.where(valid, ih_div, 0)
        safe_iw = tl.where(valid, iw_div, 0)

        n_idx_i64 = tl.full((BLOCK_M, 1), pid_n, dtype=tl.int64)
        ic_idx_i64 = ic_idx.to(tl.int64)[None, :]
        safe_ih_i64 = safe_ih.to(tl.int64)
        safe_iw_i64 = safe_iw.to(tl.int64)

        x_ptrs = (
            (((n_idx_i64 * IC) + ic_idx_i64) * IH + safe_ih_i64) * IW + safe_iw_i64
        )
        x_vals = tl.load(x_ptr + x_ptrs, mask=valid, other=0.0).to(tl.float32)

        w_mask = k_mask[:, None] & n_mask[None, :]
        n_offsets_i64 = n_offsets.to(tl.int64)[None, :]
        kh_idx_i64 = kh_idx.to(tl.int64)[:, None]
        kw_idx_i64 = kw_idx.to(tl.int64)[:, None]

        w_ptrs = (
            (((ic_idx.to(tl.int64)[:, None] * OC) + n_offsets_i64) * KH + kh_idx_i64) * KW + kw_idx_i64
        )
        w_vals = tl.load(w_ptr + w_ptrs, mask=w_mask, other=0.0).to(tl.float32)

        acc += tl.dot(x_vals, w_vals)

    if ADD_BIAS:
        bias_vals = tl.load(bias_ptr + n_offsets, mask=n_mask, other=0.0)
        acc += bias_vals[None, :]

    out_mask = m_mask[:, None] & n_mask[None, :]
    n_idx_i64 = tl.full((BLOCK_M, 1), pid_n, dtype=tl.int64)
    oh_idx_i64 = oh_idx.to(tl.int64)[:, None]
    ow_idx_i64 = ow_idx.to(tl.int64)[:, None]
    n_offsets_i64 = n_offsets.to(tl.int64)[None, :]

    y_ptrs = (
        (((n_idx_i64 * OC) + n_offsets_i64) * OH + oh_idx_i64) * OW + ow_idx_i64
    )
    tl.store(y_ptr + y_ptrs, acc.to(tl.float32), mask=out_mask)


def triton_conv_transpose2d(
    x: torch.Tensor,
    weight: torch.Tensor,
    bias: torch.Tensor,
    stride,
    padding,
    dilation,
) -> torch.Tensor:
    if isinstance(stride, tuple):
        stride_h, stride_w = stride
    else:
        stride_h = stride_w = stride
    if isinstance(padding, tuple):
        pad_h, pad_w = padding
    else:
        pad_h = pad_w = padding
    if isinstance(dilation, tuple):
        dil_h, dil_w = dilation
    else:
        dil_h = dil_w = dilation

    x = x.contiguous()
    weight = weight.contiguous()
    bias_tensor = bias.contiguous() if bias is not None else None

    N, IC, IH, IW = x.shape
    IC_w, OC, KH, KW = weight.shape
    assert IC_w == IC, "Weight/input channel mismatch."

    OH = (IH - 1) * stride_h - 2 * pad_h + dil_h * (KH - 1) + 1
    OW = (IW - 1) * stride_w - 2 * pad_w + dil_w * (KW - 1) + 1

    out = torch.empty((N, OC, OH, OW), device=x.device, dtype=x.dtype)

    K = IC * KH * KW
    BLOCK_M = 64
    BLOCK_N = 64
    BLOCK_K = 32

    bias_ptr = bias_tensor if bias_tensor is not None else out.new_empty(1)

    def grid(meta):
        return (
            N,
            triton.cdiv(OH * OW, meta["BLOCK_M"]),
            triton.cdiv(OC, meta["BLOCK_N"]),
        )

    conv_transpose2d_kernel[grid](
        x,
        weight,
        bias_ptr,
        out,
        N,
        IC,
        IH,
        IW,
        OC,
        KH,
        KW,
        OH,
        OW,
        stride_h,
        stride_w,
        pad_h,
        pad_w,
        dil_h,
        dil_w,
        K,
        ADD_BIAS=1 if bias_tensor is not None else 0,
        BLOCK_M=BLOCK_M,
        BLOCK_N=BLOCK_N,
        BLOCK_K=BLOCK_K,
        num_warps=4,
        num_stages=2,
    )

    return out


class ModelNew(nn.Module):
    def __init__(
        self,
        in_channels: int,
        out_channels: int,
        kernel_size: int,
        stride: int = 1,
        padding: int = 0,
        dilation: int = 1,
        bias: bool = False,
    ) -> None:
        super().__init__()
        self.conv = nn.ConvTranspose2d(
            in_channels,
            out_channels,
            kernel_size,
            stride=stride,
            padding=padding,
            dilation=dilation,
            bias=bias,
        )

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return triton_conv_transpose2d(
            x,
            self.conv.weight,
            self.conv.bias,
            self.conv.stride,
            self.conv.padding,
            self.conv.dilation,
        )
