import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def add_fn(a, b):
    return a + b


@triton.jit
def scan_kernel_1d(
    input_ptr,
    output_ptr,
    block_sum_ptr,
    n_rows,
    n_cols,
    stride_inp_row,
    stride_inp_col,
    stride_out_row,
    stride_out_col,
    block_sum_stride,
    BLOCK_SIZE_COL: tl.constexpr,
    BLOCK_SIZE_ROW: tl.constexpr,
):
    pid_row = tl.program_id(0)
    pid_col = tl.program_id(1)
    
    row_offset = pid_row * BLOCK_SIZE_ROW
    col_offset = pid_col * BLOCK_SIZE_COL
    
    # Iterate over rows in the block
    for r in range(BLOCK_SIZE_ROW):
        row_idx = row_offset + r
        row_active = row_idx < n_rows
        
        if row_active:
            # Compute the start of the row in the input and output
            row_start_input = row_idx * stride_inp_row
            row_start_output = row_idx * stride_out_row
            
            # Load the block for this row
            col_indices = tl.arange(0, BLOCK_SIZE_COL)
            col_offsets = col_offset + col_indices
            col_mask = col_offsets < n_cols
            
            input_ptrs = input_ptr + row_start_input + col_offsets * stride_inp_col
            block_vals = tl.load(input_ptrs, mask=col_mask, other=0.0)
            
            # Compute cumulative sum within the block for this row
            block_scan = tl.associative_scan(block_vals, 0, combine_fn=add_fn)
            
            # Store the cumulative sum for the block
            output_ptrs = output_ptr + row_start_output + col_offsets * stride_out_col
            tl.store(output_ptrs, block_scan, mask=col_mask)
            
            # Compute the sum of the block (for this row) and store it
            block_sum = tl.sum(block_vals)
            block_sum_index = row_idx * block_sum_stride + pid_col
            tl.store(block_sum_ptr + block_sum_index, block_sum)


@triton.jit
def scan_add_kernel(
    output_ptr,
    block_prefix_ptr,
    n_rows,
    n_cols,
    stride_out_row,
    stride_out_col,
    block_prefix_stride_row,
    block_prefix_stride_col,
    BLOCK_SIZE_COL: tl.constexpr,
    BLOCK_SIZE_ROW: tl.constexpr,
):
    pid_row = tl.program_id(0)
    pid_col = tl.program_id(1)
    
    if pid_col == 0:
        return
    
    row_offset = pid_row * BLOCK_SIZE_ROW
    col_offset = pid_col * BLOCK_SIZE_COL
    
    # Iterate over rows in the block
    for r in range(BLOCK_SIZE_ROW):
        row_idx = row_offset + r
        if row_idx < n_rows:
            # Load the cumulative prefix from previous blocks for this row
            prefix_ptr = block_prefix_ptr + row_idx * block_prefix_stride_row + (pid_col - 1) * block_prefix_stride_col
            prefix_val = tl.load(prefix_ptr)
            
            # Load current block values, add prefix, and store back
            col_indices = tl.arange(0, BLOCK_SIZE_COL)
            col_offsets = col_offset + col_indices
            col_mask = col_offsets < n_cols
            
            output_ptrs = output_ptr + row_idx * stride_out_row + col_offsets * stride_out_col
            curr_vals = tl.load(output_ptrs, mask=col_mask, other=0.0)
            updated_vals = curr_vals + prefix_val
            tl.store(output_ptrs, updated_vals, mask=col_mask)


def triton_cumsum(x: torch.Tensor, dim: int) -> torch.Tensor:
    if x.dim() != 2:
        raise ValueError("Input must be 2D for this implementation")
    
    if dim == 0:
        return triton_cumsum(x.t(), dim=1).t()
    
    x = x.contiguous()
    output = torch.empty_like(x)
    
    n_rows, n_cols = x.shape
    
    # Optimized block sizes for GPU memory hierarchy
    BLOCK_SIZE_COL = 2048
    BLOCK_SIZE_ROW = 2
    
    grid_cols = triton.cdiv(n_cols, BLOCK_SIZE_COL)
    grid_rows = triton.cdiv(n_rows, BLOCK_SIZE_ROW)
    
    # Store block sums
    block_sums = torch.zeros((n_rows, grid_cols), device=x.device, dtype=x.dtype)
    
    # First pass: compute cumulative sums within each block
    scan_kernel_1d[(grid_rows, grid_cols)](
        x,
        output,
        block_sums,
        n_rows,
        n_cols,
        x.stride(0),
        x.stride(1),
        output.stride(0),
        output.stride(1),
        block_sums.stride(0),
        BLOCK_SIZE_COL=BLOCK_SIZE_COL,
        BLOCK_SIZE_ROW=BLOCK_SIZE_ROW,
    )
    
    # Compute cumulative sum of block sums (prefix sums across blocks)
    if grid_cols > 1:
        block_prefix = torch.cumsum(block_sums, dim=1)
        
        # Second pass: add cumulative block sums to subsequent blocks
        scan_add_kernel[(grid_rows, grid_cols)](
            output,
            block_prefix,
            n_rows,
            n_cols,
            output.stride(0),
            output.stride(1),
            block_prefix.stride(0),
            block_prefix.stride(1),
            BLOCK_SIZE_COL=BLOCK_SIZE_COL,
            BLOCK_SIZE_ROW=BLOCK_SIZE_ROW,
        )
    
    return output
