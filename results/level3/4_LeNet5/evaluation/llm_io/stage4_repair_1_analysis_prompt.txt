def conv2d_relu_kernel(

    stride_h, stride_w,

):

    pid_n = tl.program_id(1)   # tile over flattened (B * H_out * W_out)

    offs_n = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)

    hw = offs_n % n_per_b
    ho = hw // W_out
    wo = hw % W_out

    wo_scaled = wo * stride_w

    tl.multiple_of(offs_co, BLOCK_CO)

                b * stride_xb +
