def fused_gemm_last_step_kernel(
    # Input pointers
    x_ptr,  # [B, T, K]
    weight_ptr,  # [N, K]
    bias_ptr,  # [N]

    BLOCK_K: tl.constexpr,
):
    """
    Fused kernel that:
    1. Selects the last timestep from x[:, -1, :] (shape [B, K])
    2. Performs matrix multiplication with weight [N, K]

    """

    pid_n = tl.program_id(1)  # Output feature dimension

    bias: [N] or None
