import triton.language as tl


@triton.autotune(

            num_warps=8,

            num_warps=4,
            num_stages=2,
        ),
    ],
    key=["M", "N", "K"],
)
@triton.jit
def linear_gemm_bias_kernel(
    a_ptr,        # [M, K] input
    w_ptr,        # [N, K] weight (same layout as nn.Linear.weight)
    bias_ptr,     # [N] or dummy if HAS_BIAS = False
    c_ptr,        # [M, N] output
    M, N, K,
    stride_am, stride_ak,   # strides for a_ptr: a[m, k] = a_ptr + m*stride_am + k*stride_ak
    stride_wn, stride_wk,   # strides for w_ptr: w[n, k] = w_ptr + n*stride_wn + k*stride_wk
    stride_cm, stride_cn,   # strides for c_ptr: c[m, n] = c_ptr + m*stride_cm + n*stride_cn
    HAS_BIAS: tl.constexpr,
    BLOCK_M: tl.constexpr,
    BLOCK_N: tl.constexpr,
    BLOCK_K: tl.constexpr,
    GROUP_M: tl.constexpr,
):
    # 1D program id, with grouping along M to improve L2 reuse of W tiles
    pid = tl.program_id(0)

    num_pid_m = tl.cdiv(M, BLOCK_M)
    num_pid_n = tl.cdiv(N, BLOCK_N)
    num_pid_in_group = GROUP_M * num_pid_n

    group_id = pid // num_pid_in_group
    group_pid = pid % num_pid_in_group

    pid_m = group_id * GROUP_M + (group_pid % GROUP_M)
    pid_n = group_pid // GROUP_M

    # Some pids may be out-of-range on M; guard them
    if pid_m >= num_pid_m:
        return

    # Offsets for this tile.
    offs_m = pid_m * BLOCK_M + tl.arange(0, BLOCK_M)  # [BLOCK_M]
    offs_n = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)  # [BLOCK_N]
    offs_k = tl.arange(0, BLOCK_K)                    # [BLOCK_K]

    # Masks for output tile (shared by all fused ops on C).
    m_mask = offs_m < M
    n_mask = offs_n < N
    c_mask = m_mask[:, None] & n_mask[None, :]

    # Initialize accumulator in FP32 for numerical stability.
    acc = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)

    # Base pointers to A and W tiles.
    # A tile shape: [BLOCK_M, BLOCK_K]
    a_ptrs = a_ptr + (
        offs_m[:, None] * stride_am +
        offs_k[None, :] * stride_ak
    )

    # Treat W as B = W^T with shape [K, N]:
    # B[k, n] = W[n, k]
    # so row stride_bk = stride_wk, col stride_bn = stride_wn
    w_ptrs = w_ptr + (
        offs_k[:, None] * stride_wk +
        offs_n[None, :] * stride_wn
    )

    # Main K-loop
    for k in range(0, K, BLOCK_K):
        k_offsets = k + offs_k
        k_mask = k_offsets < K

        # Load A: [BLOCK_M, BLOCK_K]
        a = tl.load(
            a_ptrs,
            mask=m_mask[:, None] & k_mask[None, :],
            other=0.0,
        )

        # Load W^T (as B): [BLOCK_K, BLOCK_N]
        w = tl.load(
            w_ptrs,
            mask=k_mask[:, None] & n_mask[None, :],
            other=0.0,
        )

        # Accumulate matrix multiply for this K block.
        acc += tl.dot(a, w, allow_tf32=True)

        # Advance to next K tile.
        a_ptrs += BLOCK_K * stride_ak
        w_ptrs += BLOCK_K * stride_wk

    # Fused bias add in-register (no intermediate stores).
    if HAS_BIAS:
        bias = tl.load(bias_ptr + offs_n, mask=n_mask, other=0.0)
        acc += bias[None, :]

    # Single store of final output tile.
    c_ptrs = c_ptr + (
        offs_m[:, None] * stride_cm +
        offs_n[None, :] * stride_cn
    )
    tl.store(c_ptrs, acc, mask=c_mask)


def triton_linear(x: torch.Tensor, weight: torch.Tensor, bias: torch.Tensor = None) -> torch.Tensor:
    """
    High-performance Linear layer using Triton GEMM + fused bias add.

    x:      [M, K]
    weight: [N, K]  (same as nn.Linear.weight: out_features, in_features)
    bias:   [N] or None
    returns [M, N]
    """
    assert x.is_cuda and weight.is_cuda, "Triton linear expects CUDA tensors"
    M, K = x.shape
    N, K_w = weight.shape
    assert K == K_w, f"Incompatible shapes: x ({M}, {K}) and weight ({N}, {K_w})"
    if bias is not None:
        assert bias.shape[0] == N, "Bias must have shape [N]"

    y = torch.empty((M, N), device=x.device, dtype=x.dtype)

    HAS_BIAS = bias is not None
    bias_ptr = bias if HAS_BIAS else x  # dummy pointer when HAS_BIAS = False

    def grid(meta):
        # 1D launch grid with grouping inside the kernel
        return (
            triton.cdiv(M, meta["BLOCK_M"]) * triton.cdiv(N, meta["BLOCK_N"]),
        )

    linear_gemm_bias_kernel[grid](
        x, weight, bias_ptr, y,
        M, N, K,
        x.stride(0), x.stride(1),
        weight.stride(0), weight.stride(1),
        y.stride(0), y.stride(1),
        HAS_BIAS=HAS_BIAS,
    )
    return y


def triton_conv1x1(x: torch.Tensor, weight: torch.Tensor) -> torch.Tensor:
    """
    1x1 convolution via Triton GEMM (no bias):
    x:      [B, C_in, H, W]
    weight: [C_out, C_in, 1, 1] (nn.Conv2d format)
    returns [B, C_out, H, W]
    """
    assert x.is_cuda and weight.is_cuda, "Triton conv1x1 expects CUDA tensors"
    B, C_in, H, W = x.shape
    C_out, C_in_w, kh, kw = weight.shape
    assert kh == 1 and kw == 1, "triton_conv1x1 only supports 1x1 kernels"
    assert C_in == C_in_w, "Input channels mismatch"

    # Reshape to 2D: [M, K] = [B*H*W, C_in]
    # NHWC contiguous then flatten to get coalesced channel loads.
    M = B * H * W
    K = C_in
    x_2d = x.permute(0, 2, 3, 1).contiguous().view(M, K)  # [M, C_in]
    # Weight already matches [C_out, C_in] = [N, K].
    w_2d = weight.view(C_out, C_in)  # [C_out, C_in]

    y_2d = triton_linear(x_2d, w_2d, bias=None)  # [M, C_out]

    # Back to [B, C_out, H, W]
    y = y_2d.view(B, H, W, C_out).permute(0, 3, 1, 2).contiguous()
    return y
