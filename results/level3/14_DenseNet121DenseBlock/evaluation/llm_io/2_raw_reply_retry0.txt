import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.autotune(
    configs=[
        # Register-pressure-aware tiles (small, square-ish, BLOCK_K=32)
        triton.Config(
            {"BLOCK_M": 64, "BLOCK_N": 64, "BLOCK_K": 32},
            num_warps=4,
            num_stages=2,
        ),
        triton.Config(
            {"BLOCK_M": 64, "BLOCK_N": 32, "BLOCK_K": 32},
            num_warps=4,
            num_stages=2,
        ),
        triton.Config(
            {"BLOCK_M": 32, "BLOCK_N": 64, "BLOCK_K": 32},
            num_warps=4,
            num_stages=2,
        ),
    ],
    key=["M", "C_out"],
)
@triton.jit
def conv2d_3x3_implicit_gemm_kernel(
    x_ptr,  # *f32, [N, C_in, H, W]
    w_ptr,  # *f32, [K, C_out] = [C_in*9, C_out]
    y_ptr,  # *f32, [N, C_out, H, W]
    N: tl.int32,
    C_in: tl.int32,
    H: tl.int32,
    W: tl.int32,
    C_out: tl.int32,
    M: tl.int32,  # N * H * W
    K: tl.int32,  # C_in * 9
    stride_xn: tl.int32,
    stride_xc: tl.int32,
    stride_xh: tl.int32,
    stride_xw: tl.int32,
    stride_wk: tl.int32,
    stride_wn: tl.int32,
    stride_yn: tl.int32,
    stride_yc: tl.int32,
    stride_yh: tl.int32,
    stride_yw: tl.int32,
    BLOCK_M: tl.constexpr,
    BLOCK_N: tl.constexpr,
    BLOCK_K: tl.constexpr,
):
    """
    Implicit-GEMM 3x3 conv:
      - Treat output as Y[M, C_out], where M = N*H*W
      - Treat input as A[M, K], where K = C_in*3*3 (im2col on the fly)
      - Treat weights as B[K, C_out] (pre-flattened and contiguous)
      - Compute Y = A @ B

    This kernel is tuned for low register pressure on Ada (SM 8.9) by:
      - Using small tiles: BLOCK_M/BLOCK_N in {32, 64}, BLOCK_K = 32
      - Avoiding complicated weight indexing (weights are [K, C_out])
    """

    # -------------------------------------------------------------------------
    # Program IDs: 2D grid over (M, C_out)
    # -------------------------------------------------------------------------
    pid_m = tl.program_id(axis=0)  # tiles over M (flattened NHW)
    pid_n = tl.program_id(axis=1)  # tiles over C_out

    offs_m = pid_m * BLOCK_M + tl.arange(0, BLOCK_M)  # [BLOCK_M]
    offs_n = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)  # [BLOCK_N]

    mask_m = offs_m < M
    mask_n = offs_n < C_out

    # Decode (n, h, w) from flattened NHW index once per tile
    HW = H * W
    n_idx = offs_m // HW
    rem_hw = offs_m - n_idx * HW
    h_idx = rem_hw // W
    w_idx = rem_hw - h_idx * W

    # Accumulator tile
    acc = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)

    # -------------------------------------------------------------------------
    # Loop over K dimension: K = C_in * 3 * 3
    # -------------------------------------------------------------------------
    k_start = 0
    while k_start < K:
        offs_k = k_start + tl.arange(0, BLOCK_K)  # [BLOCK_K]
        mask_k = offs_k < K

        # Decode (c, kh, kw) from linear k index (K = C_in * 3 * 3 = C_in * 9)
        c_idx = offs_k // 9
        rem_k = offs_k - c_idx * 9
        kh_idx = rem_k // 3
        kw_idx = rem_k - kh_idx * 3

        # Input spatial locations with padding=1
        # Shapes: h_idx/w_idx [BM], kh_idx/kw_idx [BK]
        h_in = h_idx[:, None] + kh_idx[None, :] - 1
        w_in = w_idx[:, None] + kw_idx[None, :] - 1

        in_bounds = (h_in >= 0) & (h_in < H) & (w_in >= 0) & (w_in < W)

        # Mask for input tile A
        mask_a = mask_m[:, None] & mask_k[None, :] & in_bounds

        # Input pointers X[n, c, h_in, w_in]
        x_ptrs = (
            x_ptr
            + n_idx[:, None] * stride_xn
            + c_idx[None, :] * stride_xc
            + h_in * stride_xh
            + w_in * stride_xw
        )
        a = tl.load(x_ptrs, mask=mask_a, other=0.0)  # [BM, BK]

        # Weight tile B: W[K, C_out] with simple 2D indexing
        w_ptrs = (
            w_ptr
            + offs_k[:, None] * stride_wk
            + offs_n[None, :] * stride_wn
        )
        mask_b = mask_k[:, None] & mask_n[None, :]
        b = tl.load(w_ptrs, mask=mask_b, other=0.0)  # [BK, BN]

        # Matrix multiply-accumulate
        acc += tl.dot(a, b)

        k_start += BLOCK_K

    # -------------------------------------------------------------------------
    # Store result: Y[n, oc, h, w]
    # -------------------------------------------------------------------------
    y_ptrs = (
        y_ptr
        + n_idx[:, None] * stride_yn
        + offs_n[None, :] * stride_yc
        + h_idx[:, None] * stride_yh
        + w_idx[:, None] * stride_yw
    )
    mask_y = mask_m[:, None] & mask_n[None, :]
    tl.store(y_ptrs, acc, mask=mask_y)


def conv2d_3x3_triton(x: torch.Tensor, weight: torch.Tensor) -> torch.Tensor:
    """
    3x3 conv2d with:
      - stride = 1
      - padding = 1
      - dilation = 1
      - groups = 1
      - bias = False

    Implemented as implicit-GEMM with a Triton kernel, optimized for:
      - Ada (4090, SM 8.9)
      - Low register pressure via small BLOCK_M/BLOCK_N and BLOCK_K=32
      - Simpler weight layout: W[K, C_out] = [C_in*3*3, C_out]
    """
    assert x.is_cuda and weight.is_cuda, "Inputs must be CUDA tensors"
    assert x.dtype == torch.float32 and weight.dtype == torch.float32, "Only float32 supported"
    assert x.dim() == 4 and weight.dim() == 4, "Expected 4D tensors"

    N, C_in, H, W = x.shape
    C_out, C_in_w, KH, KW = weight.shape
    assert C_in == C_in_w, "Channel mismatch between input and weight"
    assert KH == 3 and KW == 3, "Kernel size must be 3x3"

    # Make tensors contiguous for predictable strides and coalesced access
    x_c = x.contiguous()
    # Pre-flatten weights to [K, C_out] so kernel has trivial indexing
    # K = C_in * 3 * 3
    K = C_in * 3 * 3
    w_2d = weight.view(C_out, K).transpose(0, 1).contiguous()  # [K, C_out]

    y = torch.empty((N, C_out, H, W), device=x.device, dtype=x.dtype)

    stride_xn, stride_xc, stride_xh, stride_xw = x_c.stride()
    stride_wk, stride_wn = w_2d.stride()
    stride_yn, stride_yc, stride_yh, stride_yw = y.stride()

    M = N * H * W  # flattened output pixels

    def grid(meta):
        return (
            triton.cdiv(M, meta["BLOCK_M"]),
            triton.cdiv(C_out, meta["BLOCK_N"]),
        )

    conv2d_3x3_implicit_gemm_kernel[grid](
        x_c,
        w_2d,
        y,
        N,
        C_in,
        H,
        W,
        C_out,
        M,
        K,
        stride_xn,
        stride_xc,
        stride_xh,
        stride_xw,
        stride_wk,
        stride_wn,
        stride_yn,
        stride_yc,
        stride_yh,
        stride_yw,
    )

    return y


class ModelNew(nn.Module):
    def __init__(self, num_layers: int, num_input_features: int, growth_rate: int):
        """
        Triton-accelerated Dense block:
        Conv2d(3x3) layers are computed via a high-performance Triton kernel.
        """
        super(ModelNew, self).__init__()
        layers = []
        for i in range(num_layers):
            in_features = num_input_features + i * growth_rate
            layers.append(self._make_layer(in_features, growth_rate))
        self.layers = nn.ModuleList(layers)

    def _make_layer(self, in_features: int, growth_rate: int):
        """
        BatchNorm2d -> ReLU -> Conv2d(3x3, padding=1, bias=False) -> Dropout
        Conv2d weights are later used by the Triton kernel in forward().
        """
        return nn.Sequential(
            nn.BatchNorm2d(in_features),
            nn.ReLU(inplace=True),
            nn.Conv2d(in_features, growth_rate, kernel_size=3, padding=1, bias=False),
            nn.Dropout(0.0),
        )

    def forward(self, x):
        """
        :param x: (batch_size, num_input_features, height, width)
        :return: (batch_size, num_output_features, height, width)
        """
        features = [x]
        for layer in self.layers:
            bn = layer[0]
            relu = layer[1]
            conv = layer[2]
            dropout = layer[3]

            out = bn(x)
            out = relu(out)
            # Triton-accelerated 3x3 convolution (bias=False)
            out = conv2d_3x3_triton(out, conv.weight)
            if dropout.p > 0.0:
                out = torch.nn.functional.dropout(out, p=dropout.p, training=self.training)

            features.append(out)
            x = torch.cat(features, dim=1)
        return x
